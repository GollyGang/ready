101 void OpenCL_MixIn::ReloadContextIfNeeded()
102 {
103  if(!this->need_reload_context) return;
104 
105  cl_int ret;
106 
107  // retrieve our chosen platform
108  cl_platform_id platform_id;
109  {
110  const int MAX_PLATFORMS = 10;
111  cl_platform_id platforms_available[MAX_PLATFORMS];
112  cl_uint num_platforms;
113  ret = clGetPlatformIDs(MAX_PLATFORMS,platforms_available,&num_platforms);
114  if(ret != CL_SUCCESS || num_platforms==0)
115  {
116  throw runtime_error("No OpenCL platforms available");
117  // currently only likely to see this when running in a virtualized OS, where an opencl.dll is found but doesn't work
118  }
119  if(this->iPlatform >= (int)num_platforms)
120  throw runtime_error("OpenCL_MixIn::ReloadContextIfNeeded : too few platforms available");
121  platform_id = platforms_available[this->iPlatform];
122  }
123 
124  // retrieve our chosen device
125  {
126  const int MAX_DEVICES = 10;
127  cl_device_id devices_available[MAX_DEVICES];
128  cl_uint num_devices;
129  ret = clGetDeviceIDs(platform_id,CL_DEVICE_TYPE_ALL,MAX_DEVICES,devices_available,&num_devices);
130  throwOnError(ret,"OpenCL_MixIn::ReloadContextIfNeeded : Failed to retrieve device IDs: ");
131  if(this->iDevice >= (int)num_devices)
132  throw runtime_error("OpenCL_MixIn::ReloadContextIfNeeded : too few devices available");
133  this->device_id = devices_available[this->iDevice];
134  }
135 
136  // create the context
137  clReleaseContext(this->context);
138  this->context = clCreateContext(NULL,1,&this->device_id,NULL,NULL,&ret);
139  throwOnError(ret,"OpenCL_MixIn::ReloadContextIfNeeded : Failed to create context: ");
140 
141  // create the command queue
142  clReleaseCommandQueue(this->command_queue);
143  this->command_queue = clCreateCommandQueue(this->context,this->device_id,0,&ret);
144  throwOnError(ret,"OpenCL_MixIn::ReloadContextIfNeeded : Failed to create command queue: ");
145 
146  this->need_reload_context = false;
147 }

151 void OpenCL_MixIn::TestKernel(std::string kernel_source)
152 {
153  this->need_reload_context = true;
154  this->ReloadContextIfNeeded();
155 
156  cl_int ret;
157 
158  // create the program
159  const char *source = kernel_source.c_str();
160  size_t source_size = kernel_source.length();
161  cl_program temp_program = clCreateProgramWithSource(this->context,1,&source,&source_size,&ret);
162  throwOnError(ret,"OpenCL_MixIn::TestKernel : Failed to create program with source: ");
163 
164  // build the program
165  ret = clBuildProgram(temp_program,1,&this->device_id,"-cl-denorms-are-zero -cl-fast-relaxed-math",NULL,NULL);
166  if(ret != CL_SUCCESS)
167  {
168  const int MAX_BUILD_LOG = 10000;
169  char build_log[MAX_BUILD_LOG];
170  size_t build_log_length;
171  cl_int ret2 = clGetProgramBuildInfo(temp_program,this->device_id,CL_PROGRAM_BUILD_LOG,MAX_BUILD_LOG,build_log,&build_log_length);
172  throwOnError(ret2,"OpenCL_MixIn::TestKernel : retrieving program build log failed: ");
173  { ofstream out("kernel.txt"); out << kernel_source; }
174  ostringstream oss;
175  oss << "OpenCL_MixIn::TestKernel : build failed: (kernel saved as kernel.txt)\n\n" << build_log;
176  throwOnError(ret,oss.str().c_str());
177  }
178  clReleaseProgram(temp_program);
179 }
