67 void FullKernelOpenCLMeshRD::InitializeFromXML(vtkXMLDataElement *rd, bool &warn_to_update)
68 {
69  OpenCLMeshRD::InitializeFromXML(rd,warn_to_update);
70 
71  vtkSmartPointer<vtkXMLDataElement> rule = rd->FindNestedElementWithName("rule");
72  if(!rule) throw runtime_error("rule node not found in file");
73 
74  // kernel:
75  vtkSmartPointer<vtkXMLDataElement> xml_kernel = rule->FindNestedElementWithName("kernel");
76  if(!xml_kernel) throw runtime_error("kernel node not found in file");
77  string formula = trim_multiline_string(xml_kernel->GetCharacterData());
78 
79  // number_of_chemicals:
80  read_required_attribute(xml_kernel,"number_of_chemicals",this->n_chemicals);
81 
82  // do this last, because it requires everything else to be set up first
83  this->TestFormula(formula); // will throw on error but won't set
84  this->SetFormula(formula); // will set but won't throw
85 }
