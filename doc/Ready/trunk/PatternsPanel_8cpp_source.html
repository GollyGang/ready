134 void PatternsPanel::OnTreeClick(wxMouseEvent& event)
135 {
136  // determine if an item was clicked
137  wxTreeCtrl* treectrl = patternctrl->GetTreeCtrl();
138  wxPoint pt = event.GetPosition();
139  int flags;
140  wxTreeItemId id = treectrl->HitTest(pt, flags);
141  if (!id.IsOk()) {
142  // click wasn't in any item
143  event.Skip();
144  return;
145  }
146 
147  if (treectrl->ItemHasChildren(id)) {
148  // click was in a folder item
149  event.Skip();
150  return;
151  }
152 
153  // set global flag for testing in OnTreeSelChanged
154  edit_file = event.ControlDown() || event.RightDown();
155 
156  // check for click in an already selected item
157  if (id == treectrl->GetSelection()) {
158  // force a selection change so OnTreeSelChanged gets called
159  treectrl->Unselect();
160  }
161 
162  treectrl->SelectItem(id);
163  treectrl->SetFocus();
164  // OnTreeSelChanged will be called -- don't call event.Skip()
165 }
