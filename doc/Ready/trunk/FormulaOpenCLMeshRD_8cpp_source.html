32 FormulaOpenCLMeshRD::FormulaOpenCLMeshRD(int opencl_platform,int opencl_device)
33  : OpenCLMeshRD(opencl_platform,opencl_device)
34 {
35  // these settings are used in File > New Pattern
36  this->SetRuleName("Gray-Scott");
37  this->AddParameter("timestep",1.0f);
38  this->AddParameter("D_a",0.082f);
39  this->AddParameter("D_b",0.041f);
40  this->AddParameter("k",0.06f);
41  this->AddParameter("F",0.035f);
42  this->SetFormula("\
43 delta_a = D_a * laplacian_a - a*b*b + F*(1.0f-a);\n\
44 delta_b = D_b * laplacian_b + a*b*b - (F+k)*b;");

49 std::string FormulaOpenCLMeshRD::AssembleKernelSourceFromFormula(std::string f) const
50 {
51  const string indent = " ";
52  const int NC = this->GetNumberOfChemicals();
53 
54  ostringstream kernel_source;
55  kernel_source << fixed << setprecision(6);
56  // output the function definition
57  kernel_source << "__kernel void rd_compute(";
58  for(int i=0;i<NC;i++)
59  kernel_source << "__global float *" << GetChemicalName(i) << "_in,";
60  for(int i=0;i<NC;i++)
61  kernel_source << "__global float *" << GetChemicalName(i) << "_out,";
62  kernel_source << "__global int* neighbor_indices,__global float* neighbor_weights,const int max_neighbors)\n";
63  // output the body
64  kernel_source << "{\n";
65  kernel_source << indent << "const int index_x = get_global_id(0);\n";
66  for(int i=0;i<NC;i++)
67  kernel_source << indent << "float " << GetChemicalName(i) << " = " << GetChemicalName(i) << "_in[index_x];\n";
68  // compute the laplacians
69  for(int i=0;i<NC;i++)
70  kernel_source << indent << "float laplacian_" << GetChemicalName(i) << " = 0.0f;\n";
71  kernel_source << indent << "int _offset = index_x * max_neighbors;\n";
72  kernel_source << indent << "for(int _i=0;_i<max_neighbors;_i++)\n" << indent << "{\n";
73  for(int i=0;i<NC;i++)
74  kernel_source << indent << indent << "laplacian_" << GetChemicalName(i) << " += " << GetChemicalName(i)
75  << "_in[neighbor_indices[_offset+_i]] * neighbor_weights[_offset+_i];\n";
76  kernel_source << indent << "}\n";
77  for(int i=0;i<NC;i++)
78  kernel_source << indent << "laplacian_" << GetChemicalName(i) << " -= " << GetChemicalName(i) << ";\n";
79  kernel_source << indent << "// scale the Laplacians to be more similar to the 2D square grid version, so the same parameters work\n";
80  for(int i=0;i<NC;i++)
81  kernel_source << indent << "laplacian_" << GetChemicalName(i) << " *= 4.0f;\n"; // TODO: not sure about 3D meshes
82  // the parameters (assume all float for now)
83  kernel_source << "\n" << indent << "// parameters:\n";
84  for(int i=0;i<(int)this->parameters.size();i++)
85  kernel_source << indent << "float " << this->parameters[i].first << " = " << this->parameters[i].second << "f;\n";
86  // the update step
87  kernel_source << "\n" << indent << "// update step:\n";
88  for(int i=0;i<NC;i++)
89  kernel_source << indent << "float delta_" << GetChemicalName(i) << " = 0.0f;\n";
90  kernel_source << f << "\n";
91  kernel_source << "\n";
92  for(int i=0;i<NC;i++)
93  kernel_source << indent << GetChemicalName(i) << "_out[index_x] = " << GetChemicalName(i) << " + timestep * delta_" << GetChemicalName(i) << ";\n";
94  kernel_source << "}\n";
95  return kernel_source.str();
96 }
