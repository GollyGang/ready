00049 std::string FormulaOpenCLMeshRD::AssembleKernelSourceFromFormula(std::string f) const
00050 {
00051     const string indent = "    ";
00052     const int NC = this->GetNumberOfChemicals();
00053 
00054     ostringstream kernel_source;
00055     kernel_source << fixed << setprecision(6);
00056     // output the function definition
00057     kernel_source << "__kernel void rd_compute(";
00058     for(int i=0;i<NC;i++)
00059         kernel_source << "__global float *" << GetChemicalName(i) << "_in,";
00060     for(int i=0;i<NC;i++)
00061         kernel_source << "__global float *" << GetChemicalName(i) << "_out,";
00062     kernel_source << "__global int* neighbor_indices,__global float* neighbor_weights,const int max_neighbors)\n";
00063     // output the body
00064     kernel_source << "{\n";
00065     kernel_source << indent << "const int x = get_global_id(0);\n";
00066     for(int i=0;i<NC;i++)
00067         kernel_source << indent << "float " << GetChemicalName(i) << " = " << GetChemicalName(i) << "_in[x];\n";
00068     // compute the laplacians
00069     for(int i=0;i<NC;i++)
00070         kernel_source << indent << "float laplacian_" << GetChemicalName(i) << " = 0.0f;\n";
00071     kernel_source << indent << "int offset = x * max_neighbors;\n";
00072     kernel_source << indent << "for(int i=0;i<max_neighbors;i++)\n" << indent << "{\n";
00073     for(int i=0;i<NC;i++)
00074         kernel_source << indent << indent << "laplacian_" << GetChemicalName(i) << " += " << GetChemicalName(i) 
00075                       << "_in[neighbor_indices[offset+i]] * neighbor_weights[offset+i];\n";
00076     kernel_source << indent << "}\n";
00077     for(int i=0;i<NC;i++)
00078         kernel_source << indent << "laplacian_" << GetChemicalName(i) << " -= " << GetChemicalName(i) << ";\n";
00079     kernel_source << indent << "// scale the Laplacians to be more similar to the 2D square grid version, so the same parameters work\n";
00080     for(int i=0;i<NC;i++)
00081         kernel_source << indent << "laplacian_" << GetChemicalName(i) << " *= 4.0f;\n";
00082     kernel_source << indent << "// avoid denormals\n";
00083     for(int i=0;i<NC;i++)
00084         kernel_source << indent << "laplacian_" << GetChemicalName(i) << " += 1e-6f;\n";
00085     // the parameters (assume all float for now)
00086     kernel_source << "\n" << indent << "// parameters:\n";
00087     for(int i=0;i<(int)this->parameters.size();i++)
00088         kernel_source << indent << "float " << this->parameters[i].first << " = " << this->parameters[i].second << "f;\n";
00089     // the update step
00090     kernel_source << "\n" << indent << "// update step:\n";
00091     for(int i=0;i<NC;i++)
00092         kernel_source << indent << "float delta_" << GetChemicalName(i) << " = 0.0f;\n";
00093     kernel_source << f << "\n";
00094     for(int i=0;i<NC;i++)
00095         kernel_source << indent << GetChemicalName(i) << "_out[x] = " << GetChemicalName(i) << " + timestep * delta_" << GetChemicalName(i) << ";\n";
00096     kernel_source << "}\n";
