69 void GrayScottImageRD::InternalUpdate(int n_steps)
70 {
71  const int X = this->GetX();
72  const int Y = this->GetY();
73  const int Z = this->GetZ();
74 
75  float timestep = this->GetParameterValueByName("timestep");
76  float D_a = this->GetParameterValueByName("D_a");
77  float D_b = this->GetParameterValueByName("D_b");
78  float k = this->GetParameterValueByName("k");
79  float F = this->GetParameterValueByName("F");
80 
81  int x_prev,x_next,y_prev,y_next,z_prev,z_next;
82 
83  // take approximately n_steps
84  for(int iStep=0;iStep<n_steps;iStep++)
85  {
86  float *old_a,*new_a,*old_b,*new_b;
87  switch(iStep%2)
88  {
89  case 0: old_a = static_cast<float*>(this->images[0]->GetScalarPointer());
90  old_b = static_cast<float*>(this->images[1]->GetScalarPointer());
91  new_a = static_cast<float*>(this->buffer_images[0]->GetScalarPointer());
92  new_b = static_cast<float*>(this->buffer_images[1]->GetScalarPointer());
93  break;
94  case 1: old_a = static_cast<float*>(this->buffer_images[0]->GetScalarPointer());
95  old_b = static_cast<float*>(this->buffer_images[1]->GetScalarPointer());
96  new_a = static_cast<float*>(this->images[0]->GetScalarPointer());
97  new_b = static_cast<float*>(this->images[1]->GetScalarPointer());
98  break;
99  }
100  for(int z=0;z<Z;z++)
101  {
102  if(this->wrap)
103  {
104  z_prev = (z-1+Z)%Z;
105  z_next = (z+1)%Z;
106  }
107  else
108  {
109  z_prev = max(0,z-1);
110  z_next = min(Z-1,z+1);
111  }
112  for(int y=0;y<Y;y++)
113  {
114  if(this->wrap)
115  {
116  y_prev = (y-1+Y)%Y;
117  y_next = (y+1)%Y;
118  }
119  else
120  {
121  y_prev = max(0,y-1);
122  y_next = min(Y-1,y+1);
123  }
124  for(int x=0;x<X;x++)
125  {
126  if(this->wrap)
127  {
128  x_prev = (x-1+X)%X;
129  x_next = (x+1)%X;
130  }
131  else
132  {
133  x_prev = max(0,x-1);
134  x_next = min(X-1,x+1);
135  }
136 
137  float aval = *vtk_at(old_a,x,y,z,X,Y);
138  float bval = *vtk_at(old_b,x,y,z,X,Y);
139 
140  // compute the Laplacians of a and b
141  // 7-point stencil:
142  float dda = *vtk_at(old_a,x,y_prev,z,X,Y) +
143  *vtk_at(old_a,x,y_next,z,X,Y) +
144  *vtk_at(old_a,x_prev,y,z,X,Y) +
145  *vtk_at(old_a,x_next,y,z,X,Y) +
146  *vtk_at(old_a,x,y,z_prev,X,Y) +
147  *vtk_at(old_a,x,y,z_next,X,Y) - 6*aval;
148  float ddb = *vtk_at(old_b,x,y_prev,z,X,Y) +
149  *vtk_at(old_b,x,y_next,z,X,Y) +
150  *vtk_at(old_b,x_prev,y,z,X,Y) +
151  *vtk_at(old_b,x_next,y,z,X,Y) +
152  *vtk_at(old_b,x,y,z_prev,X,Y) +
153  *vtk_at(old_b,x,y,z_next,X,Y) - 6*bval;
154 
155  // compute the new rate of change of a and b
156  float da = D_a * dda - aval*bval*bval + F*(1-aval);
157  float db = D_b * ddb + aval*bval*bval - (F+k)*bval;
158 
159  #if !defined( USE_SSE )
160  // avoid denormals manually
161  da += 1e-10f;
162  db += 1e-10f;
163  #endif
164 
165  // apply the change
166  *vtk_at(new_a,x,y,z,X,Y) = aval + timestep * da;
167  *vtk_at(new_b,x,y,z,X,Y) = bval + timestep * db;
168  }
169  }
170  }
171  }
172  if(n_steps%2)
173  {
174  // output ended up in the buffer images
175  this->images[0]->DeepCopy(this->buffer_images[0]);
176  this->images[1]->DeepCopy(this->buffer_images[1]);
177  }
178 }
