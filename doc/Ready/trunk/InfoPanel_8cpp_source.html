123 void HtmlInfo::OnLinkClicked(const wxHtmlLinkInfo& link)
124 {
125  wxString url = link.GetHref();
126  if ( url.StartsWith(wxT("http:")) || url.StartsWith(wxT("mailto:")) ) {
127  // pass http/mailto URL to user's preferred browser/emailer
128  if ( !wxLaunchDefaultBrowser(url) )
129  Warning(_("Could not open URL in browser!"));
130 
131  } else if ( url.StartsWith(change_prefix) ) {
132  panel->ChangeInfo( url.Mid(change_prefix.size()) );
133  // best to reset focus after dialog closes
134  SetFocus();
135 
136  } else if ( url.StartsWith(wxT("prefs:")) ) {
137  // user clicked on link to Preferences dialog
138  frame->ShowPrefsDialog( url.AfterFirst(':') );
139 
140  } else if ( url.StartsWith(wxT("edit:")) ) {
141  // open clicked file in user's preferred text editor
142  wxString path = url.AfterFirst(':');
143  #ifdef __WXMSW__
144  path.Replace(wxT("/"), wxT("\\"));
145  #endif
146  wxFileName fname(path);
147  if (!fname.IsAbsolute()) path = readydir + path;
148  frame->EditFile(path);
149 
150 
151  } else if ( url.StartsWith(wxT("open:")) ) {
152  // open clicked file
153  wxString path = url.AfterFirst(':');
154  #ifdef __WXMSW__
155  path.Replace(wxT("/"), wxT("\\"));
156  #endif
157  wxFileName fname(path);
158  if (!fname.IsAbsolute()) path = readydir + path;
159  if (editlink) {
160  frame->EditFile(path);
161  } else {
162  frame->Raise();
163  frame->OpenFile(path);
164  }
165 
166  } else {
167  // assume it's a link to a local target
168  LoadPage(url);
169  }
170 }

361 void InfoPanel::Update(const AbstractRD* const system)
362 {
363  // build HTML string to display current parameters
364  wxString contents;
365 
366  contents += wxT("<html><body><table border=0 cellspacing=0 cellpadding=4 width=\"100%\">");
367 
368  rownum = 0;
369 
370  wxString s(system->GetRuleName().c_str(),wxConvUTF8);
371  contents += AppendRow(rule_name_label, rule_name_label, s, true);
372 
373  s = wxString(system->GetRuleType().c_str(),wxConvUTF8);
374  contents += AppendRow(rule_type_label, rule_type_label, s, false);
375 
376  s = wxString(system->GetDescription().c_str(),wxConvUTF8);
377  s.Replace(wxT("\n\n"), wxT("<p>"));
378  contents += AppendRow(description_label, description_label, s, true, true);
379 
380  contents += AppendRow(num_chemicals_label, num_chemicals_label, wxString::Format(wxT("%d"), system->GetNumberOfChemicals()),
381  system->HasEditableNumberOfChemicals());
382 
383  for(int iParam=0;iParam<(int)system->GetNumberOfParameters();iParam++)
384  {
385  contents += AppendRow(system->GetParameterName(iParam), system->GetParameterName(iParam),
386  FormatFloat(system->GetParameterValue(iParam)), true);
387  }
388 
389  wxString formula = system->GetFormula();
390  if(system->HasEditableFormula() || formula.size()>0)
391  {
392  // escape HTML reserved characters
393  formula.Replace(wxT("&"), wxT("&amp;")); // (the order of these is important)
394  formula.Replace(wxT("<"), wxT("&lt;"));
395  formula.Replace(wxT(">"), wxT("&gt;"));
396  // deal with line endings
397  formula.Replace(wxT("\r\n"), wxT("<br>"));
398  formula.Replace(wxT("\n\r"), wxT("<br>"));
399  formula.Replace(wxT("\r"), wxT("<br>"));
400  formula.Replace(wxT("\n"), wxT("<br>"));
401  // convert whitespace to &nbsp; so we can use the <code> block
402  formula.Replace(wxT(" "), wxT("&nbsp;&nbsp;"));
403  // (This is a bit of a hack. We only want to keep the leading whitespace on each line, and since &ensp; is not supported we
404  // have to use &nbsp; but this prevents wrapping. By only replacing *double* spaces we cover most usages and it's good enough for now.)
405  formula = _("<code>") + formula + _("</code>");
406  // (would prefer the <pre> block here but it adds a leading newline (which we can't use CSS to get rid of) and also prevents wrapping)
407  if(system->GetRuleType()=="kernel")
408  contents += AppendRow(kernel_label, kernel_label, formula, system->HasEditableFormula(), true);
409  else
410  contents += AppendRow(formula_label, formula_label, formula, system->HasEditableFormula(), true);
411  }
412 
413  if(system->HasEditableDimensions())
414  contents += AppendRow(dimensions_label, dimensions_label, wxString::Format(wxT("%s x %s x %s"),
415  FormatFloat(system->GetX(),3),FormatFloat(system->GetY(),3),FormatFloat(system->GetZ(),3)),
416  system->HasEditableDimensions());
417  else
418  contents += AppendRow(number_of_cells_label, number_of_cells_label, wxString::Format(wxT("%d"),
419  system->GetNumberOfCells()),false);
420 
421  if(!system->HasEditableDimensions() || system->GetRuleType()!="kernel")
422  {
423  // (hide the neighborhood for image-based kernels, which don't use it)
424  contents += AppendRow(neighborhood_type_label, neighborhood_type_label, system->GetNeighborhoodType()+"-neighbors", false);
425  contents += AppendRow(neighborhood_range_label, neighborhood_range_label, wxString::Format(wxT("%d"),system->GetNeighborhoodRange()), false);
426  contents += AppendRow(neighborhood_weight_label, neighborhood_weight_label, system->GetNeighborhoodWeight(), false);
427  }
428 
429  /* bit technical, leave as a file-only option
430  contents += AppendRow(block_size_label, block_size_label, wxString::Format(wxT("%d x %d x %d"),
431  system->GetBlockSizeX(),system->GetBlockSizeY(),system->GetBlockSizeZ()),
432  system->HasEditableBlockSize());*/
433 
434  if(system->HasEditableWrapOption())
435  contents += AppendRow(wrap_label, wrap_label, system->GetWrap()?_("on"):_("off"), true);
436 
437  contents += _T("</table>");
438 
439  contents += wxT("<h5><center>");
440  contents += _("Render settings:");
441  contents += wxT("</h5></center>");
442  contents += wxT("<table border=0 cellspacing=0 cellpadding=4 width=\"100%\">");
443 
444  rownum = 1; // nicer if 1st render setting has gray background
445 
446  const Properties& render_settings = frame->GetRenderSettings();
447  for(int i=0;i<render_settings.GetNumberOfProperties();i++)
448  {
449  const Property& prop = render_settings.GetProperty(i);
450  string name = prop.GetName();
451  wxString print_label(name);
452  print_label.Replace(_T("_"),_T(" "));
453  string type = prop.GetType();
454  if(type=="float")
455  contents += AppendRow(print_label, name, FormatFloat(prop.GetFloat()), true);
456  else if(type=="bool")
457  contents += AppendRow(print_label, name, prop.GetBool()?_("true"):_("false"), true);
458  else if(type=="int")
459  contents += AppendRow(print_label, name, FormatFloat(prop.GetInt()), true);
460  else if(type=="color")
461  {
462  float r,g,b;
463  prop.GetColor(r,g,b);
464  wxColor col(r*255,g*255,b*255);
465  contents += AppendRow(print_label, name, _("RGB = ")+FormatFloat(r,2)+_T(", ")+FormatFloat(g,2)+_T(", ")+FormatFloat(b,2),
466  true, false, col.GetAsString(wxC2S_HTML_SYNTAX));
467  }
468  else if(type=="chemical")
469  contents += AppendRow(print_label, name, prop.GetChemical(), true);
470  else if(type=="axis")
471  contents += AppendRow(print_label, name, prop.GetAxis(), true);
472  else throw runtime_error("InfoPanel::Update : unrecognised type: "+type);
473  }
474 
475  contents += _T("</table></body></html>");
476 
477  html->SaveScrollPos();
478  html->Freeze(); // prevent flicker
479  html->SetPage(contents);
480  html->RestoreScrollPos();
481  html->Thaw();
482 }

486 wxString InfoPanel::AppendRow(const wxString& print_label, const wxString& label, const wxString& value,
487  bool is_editable, bool is_multiline, const wxString& color)
488 {
489  wxString result = (rownum & 1) ? _T("<tr bgcolor=\"#F0F0F0\">") : _T("<tr>");
490  rownum++;
491 
492  if (is_multiline) {
493  result += _T("<td width=3></td><td valign=top width=\"45%\"><b>");
494  result += print_label;
495  result += _T("</b></td><td valign=top></td>");
496  // see below for how value is formatted
497  } else {
498  result += _T("<td width=3></td><td valign=top width=\"45%\"><b>");
499  result += print_label;
500  result += _T("</b></td><td valign=top>");
501  if (color.empty()) {
502  result += value;
503  } else {
504  // start with a color block to illustrate the color
505  // (we put a border around block in case color matches row background)
506  result += _T("<table border=0 cellspacing=0 cellpadding=0><tr><td>");
507  result += _T("<table border=1 cellspacing=0 cellpadding=6><tr bgcolor=\"");
508  result += color;
509  result += _T("\"><td width=25></td></tr></table></td><td valign=top>&nbsp;&nbsp;&nbsp;");
510  result += value;
511  result += _T("</td></tr></table>");
512  }
513  result += _T("</td>");
514  }
515 
516  if (is_editable) {
517  result += _T("<td valign=top align=right><a href=\"");
518  result += change_prefix;
519  result += label;
520  result += _T("\">");
521  result += _("edit");
522  result += _T("</a></td>");
523  } else {
524  result += _T("<td></td>");
525  }
526  result += _T("<td width=3></td></tr>");
527 
528  if (is_multiline) {
529  // put text in a new row that spans all columns
530  result += (rownum & 1) ? _T("<tr>") : _T("<tr bgcolor=\"#F0F0F0\">");
531  result += _T("<td width=3></td><td colspan=3>");
532  result += value;
533  result += _T("<br></td><td width=3></td></tr>");
534  }
535 
536  return result;
537 }

935  }

938  frame->ProcessKey(key, mods);
939  return true;
