00120 void HtmlInfo::OnLinkClicked(const wxHtmlLinkInfo& link)
00121 {
00122     wxString url = link.GetHref();
00123     if ( url.StartsWith(wxT("http:")) || url.StartsWith(wxT("mailto:")) ) {
00124         // pass http/mailto URL to user's preferred browser/emailer
00125         if ( !wxLaunchDefaultBrowser(url) )
00126             Warning(_("Could not open URL in browser!"));
00127 
00128     } else if ( url.StartsWith(change_prefix) ) {
00129         panel->ChangeInfo( url.Mid(change_prefix.size()) );
00130         // best to reset focus after dialog closes
00131         SetFocus();
00132 
00133     } else if ( url.StartsWith(wxT("prefs:")) ) {
00134         // user clicked on link to Preferences dialog
00135         frame->ShowPrefsDialog( url.AfterFirst(':') );
00136 
00137     } else if ( url.StartsWith(wxT("edit:")) ) {
00138         // open clicked file in user's preferred text editor
00139         wxString path = url.AfterFirst(':');
00140         #ifdef __WXMSW__
00141             path.Replace(wxT("/"), wxT("\\"));
00142         #endif
00143         wxFileName fname(path);
00144         if (!fname.IsAbsolute()) path = readydir + path;
00145         frame->EditFile(path);
00146 
00147 
00148     } else if ( url.StartsWith(wxT("open:")) ) {
00149         // open clicked file
00150         wxString path = url.AfterFirst(':');
00151         #ifdef __WXMSW__
00152             path.Replace(wxT("/"), wxT("\\"));
00153         #endif
00154         wxFileName fname(path);
00155         if (!fname.IsAbsolute()) path = readydir + path;
00156         if (editlink) {
00157             frame->EditFile(path);
00158         } else {
00159             frame->Raise();
