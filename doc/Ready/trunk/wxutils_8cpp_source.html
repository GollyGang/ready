124 bool IsHTMLFile(const wxString& filename)
125 {
126  wxString ext = filename.AfterLast('.');
127  // if filename has no extension then ext == filename
128  if (ext == filename) return false;
129  return ( ext.IsSameAs(wxT("htm"),false) ||
130  ext.IsSameAs(wxT("html"),false) );

135 bool IsTextFile(const wxString& filename)
136 {
137  if (!IsHTMLFile(filename)) {
138  // if non-html file name contains "readme" then assume it's a text file
139  wxString name = filename.AfterLast(wxFILE_SEP_PATH).MakeLower();
140  if (name.Contains(wxT("readme"))) return true;
141  }
142  wxString ext = filename.AfterLast('.');
143  // if filename has no extension then ext == filename
144  if (ext == filename) return false;
145  return ( ext.IsSameAs(wxT("txt"),false) ||
146  ext.IsSameAs(wxT("doc"),false) );
147 }

212 wxString FormatFloat(float f,int mdp)
213 {
214  wxString result = wxString::Format(wxT("%.*f"),mdp,f);
215  // strip any trailing zeros
216  while (result.GetChar(result.Length()-1) == wxChar('0')) {
217  result.Truncate(result.Length()-1);
218  }
219  // strip any trailing decimal separator
220  if (result.GetChar(result.Length()-1) == wxChar('.') || result.GetChar(result.Length()-1) == wxChar(',')) {
221  result.Truncate(result.Length()-1);
222  }
223  return result;
224 }
