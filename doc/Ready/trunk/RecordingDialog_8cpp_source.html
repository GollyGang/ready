100 bool RecordingDialog::Validate()
101 {
102  if( !wxDirExists(this->folder_edit->GetValue()) )
103  {
104  int answer = wxMessageBox("Folder doesn't exist. Create?","Confirm",wxOK|wxCANCEL);
105  if( answer == wxOK )
106  {
107  bool ret = wxFileName::Mkdir(this->folder_edit->GetValue(),wxS_DIR_DEFAULT,wxPATH_MKDIR_FULL);
108  if( !ret )
109  {
110  wxMessageBox("Unable to create folder.","Error",wxOK|wxICON_ERROR);
111  return false;
112  }
113  }
114  else
115  {
116  return false;
117  }
118  }
119  return wxDialog::Validate();
120 }
