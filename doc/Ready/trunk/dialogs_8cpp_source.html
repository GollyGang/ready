488 {
489  int key = event.GetKeyCode();
490 
491  if (event.CmdDown()) {
492  // allow handling of cmd-x/v/etc
493  event.Skip();
494 
495  } else if ( key == WXK_TAB ) {
496  wxSpinCtrl* sc = (wxSpinCtrl*) FindWindowById(ID_SPIN_CTRL);
497  if ( sc ) {
498  sc->SetFocus();
499  sc->SetSelection(-1,-1);

501

503  if ( (key >= '0' && key <= '9') || key == '+' || key == '-' ) {

506  } else {
