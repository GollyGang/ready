00340 int SplitEdge(const Tri &tri,int i1,int i2,double &x, double &y,TPairIndex &edge_splits,vtkPoints* pts)

00343 
00344     x = tri.p[i1][0] + (tri.p[i2][0] - tri.p[i1][0]) / goldenRatio;

00352         return found->second;
00353     }
00354     else
00355     {
00356         int iP = pts->InsertNextPoint(x,y,0);

00823 void MeshGenerators::GetDiamondCells(int side,vtkUnstructuredGrid *mesh,int n_chems)
00824 {
00825     // a triakis truncated tetrahedron: 16 points, 4 hexagonal faces, 12 triangular faces
00826     const double RR2 = 1.0 / sqrt(2.0); // reciprocal of root 2
00827     const double third = 1.0 / 3.0;
00828     const double coords[16][3] = { 
00829         //{-1,0,-RR2}, {1,0,-RR2}, {0,-1,RR2}, {0,1,RR2},          // (tips of the original tetrahedron)
00830         {-2*third,0,-2*third*RR2}, {2*third,0,-2*third*RR2},       // 0, 1
00831         {0,-2*third,2*third*RR2}, {0,2*third,2*third*RR2},         // 2, 3 (centroids of the snipped-off tetrahedra)
00832         {-third,0,-RR2}, {third,0,-RR2},                           // 4, 5
00833         {0,-third,RR2}, {0,third,RR2},                             // 6, 7
00834         {2*third,-third,-third*RR2},  {third,-2*third,third*RR2},  // 8, 9
00835         {2*third,third,-third*RR2},   {third,2*third,third*RR2},   // 10, 11
00836         {-2*third,-third,-third*RR2}, {-third,-2*third,third*RR2}, // 12, 13
00837         {-2*third,third,-third*RR2},  {-third,2*third,third*RR2}   // 14, 15
00838     };

00840     const int tri_faces[12][3] = { {0,12,14},{0,14,4},{0,4,12}, {1,10,8},{1,8,5},{1,5,10},
00841         {2,9,6},{2,6,13},{2,13,9}, {3,7,11},{3,11,15},{3,15,7} };
00842

00844     vtkSmartPointer<vtkAppendFilter> append = vtkSmartPointer<vtkAppendFilter>::New();

00846 
00847     for(int x=0;x<side;x++)
00848     {

00851             for(int z=0;z<side*2;z++)

00856                 for(int i=0;i<16;i++) 
00857                 {
00858                     double offset[3] = { x*(1+third), y*(1+third), (z/2)*RR2*(1+third) + (z%2)*2*third*RR2 };
