49 AbstractRD* SystemFactory::CreateFromFile(const char *filename,bool is_opencl_available,int opencl_platform,int opencl_device,
50  Properties &render_settings,bool &warn_to_update)
51 {
52  // temporarily turn off internationalisation, to avoid string-to-float conversion issues
53  char *old_locale = setlocale(LC_NUMERIC,"C");
54 
55  vtkSmartPointer<vtkXMLGenericDataObjectReader> generic_reader = vtkSmartPointer<vtkXMLGenericDataObjectReader>::New();
56  bool parallel;
57  int data_type = generic_reader->ReadOutputType(filename,parallel);
58  AbstractRD *system;
59  switch(data_type)
60  {
61  case VTK_IMAGE_DATA:
62  system = CreateFromImageDataFile(filename,is_opencl_available,opencl_platform,opencl_device,
63  render_settings,warn_to_update);
64  break;
65  case VTK_UNSTRUCTURED_GRID:
66  system = CreateFromUnstructuredGridFile(filename,is_opencl_available,opencl_platform,opencl_device,
67  render_settings,warn_to_update);
68  break;
69  default:
70  throw runtime_error("Unsupported data type or file read error");
71  }
72 
73  // restore the old locale
74  setlocale(LC_NUMERIC,old_locale);
75 
76  system->SetFilename(filename);
77  system->SetModified(false);
78  return system;
79 }

83 AbstractRD* CreateFromImageDataFile(const char *filename,bool is_opencl_available,int opencl_platform,int opencl_device,
84  Properties &render_settings,bool &warn_to_update)
85 {
86  vtkSmartPointer<RD_XMLImageReader> reader = vtkSmartPointer<RD_XMLImageReader>::New();
87  reader->SetFileName(filename);
88  reader->Update();
89  vtkImageData *image = reader->GetOutput();
90 
91  string type = reader->GetType();
92  string name = reader->GetName();
93 
94  ImageRD* image_system;
95  if(type=="inbuilt")
96  {
97  if(name=="Gray-Scott")
98  image_system = new GrayScottImageRD();
99  else
100  throw runtime_error("Unsupported inbuilt implementation: "+name);
101  }
102  else if(type=="formula")
103  {
104  if(!is_opencl_available)
105  throw runtime_error("Pattern requires OpenCL");
106  image_system = new FormulaOpenCLImageRD(opencl_platform,opencl_device);
107  }
108  else if(type=="kernel")
109  {
110  if(!is_opencl_available)
111  throw runtime_error("Pattern requires OpenCL");
112  image_system = new FullKernelOpenCLImageRD(opencl_platform,opencl_device);
113  }
114  else throw runtime_error("Unsupported rule type: "+type);
115  image_system->InitializeFromXML(reader->GetRDElement(),warn_to_update);
116 
117  // render settings
118  vtkSmartPointer<vtkXMLDataElement> xml_render_settings =
119  reader->GetRDElement()->FindNestedElementWithName("render_settings");
120  if(xml_render_settings) // optional
121  render_settings.OverwriteFromXML(xml_render_settings);
122 
123  int dim[3];
124  image->GetDimensions(dim);
125  int nc = image->GetNumberOfScalarComponents() * image->GetPointData()->GetNumberOfArrays();
126  image_system->SetDimensions(dim[0],dim[1],dim[2]);
127  image_system->SetNumberOfChemicals(nc);
128  if(reader->ShouldGenerateInitialPatternWhenLoading())
129  image_system->GenerateInitialPattern();
130  else
131  image_system->CopyFromImage(image);
132 
133  return image_system;
134 }
