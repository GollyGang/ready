45 void OpenCLImageRD::ReloadKernelIfNeeded()
46 {
47  if(!this->need_reload_formula) return;
48 
49  cl_int ret;
50 
51  // create the program
52  this->kernel_source = this->AssembleKernelSourceFromFormula(this->formula);
53  const char *source = this->kernel_source.c_str();
54  size_t source_size = this->kernel_source.length();
55  clReleaseProgram(this->program);
56  this->program = clCreateProgramWithSource(this->context,1,&source,&source_size,&ret);
57  throwOnError(ret,"OpenCLImageRD::ReloadKernelIfNeeded : Failed to create program with source: ");
58 
59  // build the program
60  ret = clBuildProgram(this->program,1,&this->device_id,"-cl-denorms-are-zero -cl-fast-relaxed-math",NULL,NULL);
61  if(ret != CL_SUCCESS)
62  {
63  const int MAX_BUILD_LOG = 10000;
64  char build_log[MAX_BUILD_LOG];
65  size_t build_log_length;
66  cl_int ret2 = clGetProgramBuildInfo(this->program,this->device_id,CL_PROGRAM_BUILD_LOG,MAX_BUILD_LOG,build_log,&build_log_length);
67  throwOnError(ret2,"OpenCLImageRD::ReloadKernelIfNeeded : retrieving program build log failed: ");
68  { ofstream out("kernel.txt"); out << kernel_source; }
69  ostringstream oss;
70  oss << "OpenCLImageRD::ReloadKernelIfNeeded : build failed (kernel saved as kernel.txt):\n\n" << build_log;
71  throwOnError(ret,oss.str().c_str());
72  }
73 
74  // create the kernel
75  clReleaseKernel(this->kernel);
76  this->kernel = clCreateKernel(this->program,this->kernel_function_name.c_str(),&ret);
77  throwOnError(ret,"OpenCLImageRD::ReloadKernelIfNeeded : kernel creation failed: ");
78 
79  this->global_range[0] = max(1,vtkMath::Round(this->GetX()) / this->GetBlockSizeX());
80  this->global_range[1] = max(1,vtkMath::Round(this->GetY()) / this->GetBlockSizeY());
81  this->global_range[2] = max(1,vtkMath::Round(this->GetZ()) / this->GetBlockSizeZ());
82  // (we let the local work group size be automatically decided, seems to be faster and more flexible that way)
83 
84  this->need_reload_formula = false;
85 }

89 void OpenCLImageRD::CreateOpenCLBuffers()
90 {
91  this->ReloadContextIfNeeded();
92 
93  const unsigned long MEM_SIZE = sizeof(float) * this->GetX() * this->GetY() * this->GetZ();
94  const int NC = this->GetNumberOfChemicals();
95 
96  this->ReleaseOpenCLBuffers();
97 
98  cl_int ret;
99 
100  for(int io=0;io<2;io++) // we create two buffers for each chemical, and switch between them
101  {
102  this->buffers[io].resize(NC);
103  for(int ic=0;ic<NC;ic++)
104  {
105  this->buffers[io][ic] = clCreateBuffer(this->context, CL_MEM_READ_WRITE, MEM_SIZE, NULL, &ret);
106  throwOnError(ret,"OpenCLImageRD::CreateOpenCLBuffers : buffer creation failed: ");
107  }
108  }
109 }

113 void OpenCLImageRD::WriteToOpenCLBuffersIfNeeded()
114 {
115  if(!this->need_write_to_opencl_buffers) return;
116 
117  const unsigned long MEM_SIZE = sizeof(float) * this->GetX() * this->GetY() * this->GetZ();
118 
119  this->iCurrentBuffer = 0;
120  for(int ic=0;ic<this->GetNumberOfChemicals();ic++)
121  {
122  float* data = static_cast<float*>(this->images[ic]->GetScalarPointer());
123  cl_int ret = clEnqueueWriteBuffer(this->command_queue,this->buffers[this->iCurrentBuffer][ic], CL_TRUE, 0, MEM_SIZE, data, 0, NULL, NULL);
124  throwOnError(ret,"OpenCLImageRD::WriteToOpenCLBuffers : buffer writing failed: ");
125  }
126 
127  this->need_write_to_opencl_buffers = false;
128 }

156 void OpenCLImageRD::AllocateImages(int x,int y,int z,int nc)
157 {
158  if(x&(x-1) || y&(y-1) || z&(z-1))

170 void OpenCLImageRD::InternalUpdate(int n_steps)
171 {
172  this->ReloadContextIfNeeded();
173  this->ReloadKernelIfNeeded();
174  this->WriteToOpenCLBuffersIfNeeded();
175 
176  cl_int ret;
177  int iBuffer;
178  const int NC = this->GetNumberOfChemicals();
179 
180  for(int it=0;it<n_steps;it++)
181  {
182  for(int io=0;io<2;io++) // first input buffers (io=0) then output buffers (io=1)
183  {
184  iBuffer = (this->iCurrentBuffer+io)%2;
185  for(int ic=0;ic<NC;ic++)
186  {
187  // a_in, b_in, ... a_out, b_out ...
188  ret = clSetKernelArg(this->kernel, io*NC+ic, sizeof(cl_mem), (void *)&this->buffers[iBuffer][ic]);
189  throwOnError(ret,"OpenCLImageRD::InternalUpdate : clSetKernelArg failed: ");
190  }
191  }
192  ret = clEnqueueNDRangeKernel(this->command_queue,this->kernel, 3, NULL, this->global_range, NULL, 0, NULL, NULL);
193  throwOnError(ret,"OpenCLImageRD::InternalUpdate : clEnqueueNDRangeKernel failed: ");
194  this->iCurrentBuffer = 1 - this->iCurrentBuffer;
195  }
196 
197  this->ReadFromOpenCLBuffers();
198 }
