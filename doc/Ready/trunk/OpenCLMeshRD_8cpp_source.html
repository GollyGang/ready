00086     for(int it=0;it<n_steps;it++)

00099         throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clEnqueueNDRangeKernel failed: ");
00100         this->iCurrentBuffer = 1 - this->iCurrentBuffer;
00101     }
00102 
00103     this->ReadFromOpenCLBuffers();

00156 void OpenCLMeshRD::CreateOpenCLBuffers()
00157 {
00158     const unsigned long MEM_SIZE = sizeof(float) * (unsigned long)this->mesh->GetNumberOfCells();
00159     const int NC = this->GetNumberOfChemicals();
00160 
00161     cl_int ret;
00162 
00163     // create two buffers for each chemical (we will switch between them)
00164     for(int io=0;io<2;io++)
00165     {
00166         this->buffers[io].resize(NC);
00167         for(int ic=0;ic<NC;ic++)
00168         {
00169             this->buffers[io][ic] = clCreateBuffer(this->context, CL_MEM_READ_WRITE, MEM_SIZE, NULL, &ret);
00170             throwOnError(ret,"OpenCLMeshRD::CreateOpenCLBuffers : buffer creation failed: ");
00171         }
00172     }
00173     
00174     // create a buffer for the indices of the neighbors of each cell
00175     const unsigned long NBORS_INDICES_SIZE = sizeof(int) * (unsigned long)this->mesh->GetNumberOfCells() * this->max_neighbors;
00176     this->clBuffer_cell_neighbor_indices = clCreateBuffer(this->context, CL_MEM_READ_ONLY, NBORS_INDICES_SIZE, NULL, &ret);
00177     throwOnError(ret,"OpenCLMeshRD::CreateOpenCLBuffers : buffer creation failed: ");
00178 
00179     // create a buffer for the diffusion coefficients of the neighbors of each cell
00180     const unsigned long NBORS_WEIGHTS_SIZE = sizeof(float) * (unsigned long)this->mesh->GetNumberOfCells() * this->max_neighbors;
00181     this->clBuffer_cell_neighbor_weights = clCreateBuffer(this->context, CL_MEM_READ_ONLY, NBORS_WEIGHTS_SIZE, NULL, &ret);
00182     throwOnError(ret,"OpenCLMeshRD::CreateOpenCLBuffers : buffer creation failed: ");
00183 
00184     // pass the neighbor indices and weights as parameters for the kernel
00185     ret = clSetKernelArg(this->kernel, 2*NC + 0, sizeof(cl_mem), (void *)&this->clBuffer_cell_neighbor_indices);
00186     throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on indices array: ");
00187     ret = clSetKernelArg(this->kernel, 2*NC + 1, sizeof(cl_mem), (void *)&this->clBuffer_cell_neighbor_weights);
00188     throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on weights array: ");
00189     ret = clSetKernelArg(this->kernel, 2*NC + 2, sizeof(int), &this->max_neighbors);
00190     throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on max_neighbors parameter: ");
00191 }
