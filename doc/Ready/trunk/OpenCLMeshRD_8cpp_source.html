103 void OpenCLMeshRD::InternalUpdate(int n_steps)
104 {
105  this->ReloadContextIfNeeded();
106  this->ReloadKernelIfNeeded();
107  this->WriteToOpenCLBuffersIfNeeded();
108 
109  cl_int ret;
110  int iBuffer;
111  const int NC = this->GetNumberOfChemicals();
112 
113  // pass the neighbor indices and weights as parameters for the kernel
114  ret = clSetKernelArg(this->kernel, 2*NC + 0, sizeof(cl_mem), (void *)&this->clBuffer_cell_neighbor_indices);
115  throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on indices array: ");
116  ret = clSetKernelArg(this->kernel, 2*NC + 1, sizeof(cl_mem), (void *)&this->clBuffer_cell_neighbor_weights);
117  throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on weights array: ");
118  ret = clSetKernelArg(this->kernel, 2*NC + 2, sizeof(int), &this->max_neighbors);
119  throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on max_neighbors parameter: ");
120 
121  for(int it=0;it<n_steps;it++)
122  {
123  for(int io=0;io<2;io++) // first input buffers (io=0) then output buffers (io=1)
124  {
125  iBuffer = (this->iCurrentBuffer+io)%2;
126  for(int ic=0;ic<NC;ic++)
127  {
128  // a_in, b_in, ... a_out, b_out ...
129  ret = clSetKernelArg(this->kernel, io*NC+ic, sizeof(cl_mem), &this->buffers[iBuffer][ic]);
130  throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clSetKernelArg failed on buffer: ");
131  }
132  }
133  ret = clEnqueueNDRangeKernel(this->command_queue,this->kernel, 3, NULL, this->global_range, NULL, 0, NULL, NULL);
134  throwOnError(ret,"OpenCLMeshRD::InternalUpdate : clEnqueueNDRangeKernel failed: ");
135  this->iCurrentBuffer = 1 - this->iCurrentBuffer;
136  }
137 
138  this->ReadFromOpenCLBuffers();
139 }

143 void OpenCLMeshRD::ReloadKernelIfNeeded()
144 {
145  if(!this->need_reload_formula) return;
146 
147  if(this->n_chemicals==0)
148  throw runtime_error("OpenCLMeshRD::ReloadKernelIfNeeded : zero chemicals");
149 
150  cl_int ret;
151 
152  // create the program
153  this->kernel_source = this->AssembleKernelSourceFromFormula(this->formula);
154  const char *source = this->kernel_source.c_str();
155  size_t source_size = this->kernel_source.length();
156  clReleaseProgram(this->program);
157  this->program = clCreateProgramWithSource(this->context,1,&source,&source_size,&ret);
158  throwOnError(ret,"OpenCLMeshRD::ReloadKernelIfNeeded : Failed to create program with source: ");
159 
160  // build the program
161  ret = clBuildProgram(this->program,1,&this->device_id,"-cl-denorms-are-zero -cl-fast-relaxed-math",NULL,NULL);
162  if(ret != CL_SUCCESS)
163  {
164  const int MAX_BUILD_LOG = 10000;
165  char build_log[MAX_BUILD_LOG];
166  size_t build_log_length;
167  cl_int ret2 = clGetProgramBuildInfo(this->program,this->device_id,CL_PROGRAM_BUILD_LOG,MAX_BUILD_LOG,build_log,&build_log_length);
168  throwOnError(ret2,"OpenCLMeshRD::ReloadKernelIfNeeded : retrieving program build log failed: ");
169  { ofstream out("kernel.txt"); out << kernel_source; }
170  ostringstream oss;
171  oss << "OpenCLMeshRD::ReloadKernelIfNeeded : build failed (kernel saved as kernel.txt):\n\n" << build_log;
172  throwOnError(ret,oss.str().c_str());
173  }
174 
175  // create the kernel
176  clReleaseKernel(this->kernel);
177  this->kernel = clCreateKernel(this->program,this->kernel_function_name.c_str(),&ret);
178  throwOnError(ret,"OpenCLMeshRD::ReloadKernelIfNeeded : kernel creation failed: ");
179 
180  // TODO: round this up to an abundant number to enable many choices for division by local workgroup range?
181  this->global_range[0] = this->mesh->GetNumberOfCells();
182  this->global_range[1] = 1;
183  this->global_range[2] = 1;
184  // (we let the local work group size be automatically decided, seems to be faster and more flexible that way)
185 
186  this->need_reload_formula = false;
187 }

191 void OpenCLMeshRD::CreateOpenCLBuffers()
192 {
193  this->ReloadContextIfNeeded();
194 
195  const unsigned long MEM_SIZE = sizeof(float) * (unsigned long)this->mesh->GetNumberOfCells();
196  const int NC = this->GetNumberOfChemicals();
197 
198  this->ReleaseOpenCLBuffers();
199 
200  cl_int ret;
201 
202  // create two buffers for each chemical (we will switch between them)
203  for(int io=0;io<2;io++)
204  {
205  this->buffers[io].resize(NC);
206  for(int ic=0;ic<NC;ic++)
207  {
208  this->buffers[io][ic] = clCreateBuffer(this->context, CL_MEM_READ_WRITE, MEM_SIZE, NULL, &ret);
209  throwOnError(ret,"OpenCLMeshRD::CreateOpenCLBuffers : buffer creation failed: ");
210  }
211  }
212 
213  // create a buffer for the indices of the neighbors of each cell
214  const unsigned long NBORS_INDICES_SIZE = sizeof(int) * (unsigned long)this->mesh->GetNumberOfCells() * this->max_neighbors;
215  this->clBuffer_cell_neighbor_indices = clCreateBuffer(this->context, CL_MEM_READ_ONLY, NBORS_INDICES_SIZE, NULL, &ret);
216  throwOnError(ret,"OpenCLMeshRD::CreateOpenCLBuffers : neighbor_indices buffer creation failed: ");
217 
218  // create a buffer for the diffusion coefficients of the neighbors of each cell
219  const unsigned long NBORS_WEIGHTS_SIZE = sizeof(float) * (unsigned long)this->mesh->GetNumberOfCells() * this->max_neighbors;
220  this->clBuffer_cell_neighbor_weights = clCreateBuffer(this->context, CL_MEM_READ_ONLY, NBORS_WEIGHTS_SIZE, NULL, &ret);
221  throwOnError(ret,"OpenCLMeshRD::CreateOpenCLBuffers : neighbor_weights buffer creation failed: ");
222 }

226 void OpenCLMeshRD::WriteToOpenCLBuffersIfNeeded()
227 {
228  if(!this->need_write_to_opencl_buffers) return;
229 
230  if(this->buffers[0].empty())
231  this->CreateOpenCLBuffers();
232 
233  cl_int ret;
234  const unsigned long MEM_SIZE = sizeof(float) * (int)this->mesh->GetNumberOfCells();
235  this->iCurrentBuffer = 0;
236  for(int ic=0;ic<this->GetNumberOfChemicals();ic++)
237  {
238  float* data = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(ic).c_str()) )->GetPointer(0);
239  ret = clEnqueueWriteBuffer(this->command_queue,this->buffers[this->iCurrentBuffer][ic], CL_TRUE, 0, MEM_SIZE, data, 0, NULL, NULL);
240  throwOnError(ret,"OpenCLMeshRD::WriteToOpenCLBuffers : buffer writing failed: ");
241  }
242 
243  // fill indices buffer
244  const unsigned long NBORS_INDICES_SIZE = sizeof(int) * (unsigned long)this->mesh->GetNumberOfCells() * this->max_neighbors;
245  ret = clEnqueueWriteBuffer(this->command_queue,this->clBuffer_cell_neighbor_indices, CL_TRUE, 0, NBORS_INDICES_SIZE, this->cell_neighbor_indices, 0, NULL, NULL);
246  throwOnError(ret,"OpenCLMeshRD::WriteToOpenCLBuffers : buffer writing failed: ");
247 
248  // fill weights buffer
249  const unsigned long NBORS_WEIGHTS_SIZE = sizeof(float) * (unsigned long)this->mesh->GetNumberOfCells() * this->max_neighbors;
250  ret = clEnqueueWriteBuffer(this->command_queue,this->clBuffer_cell_neighbor_weights, CL_TRUE, 0, NBORS_WEIGHTS_SIZE, this->cell_neighbor_weights, 0, NULL, NULL);
251  throwOnError(ret,"OpenCLMeshRD::WriteToOpenCLBuffers : buffer writing failed: ");
252 
253  this->need_write_to_opencl_buffers = false;
254 }
