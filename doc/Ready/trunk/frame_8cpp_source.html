1024 void MyFrame::OnStep(wxCommandEvent& event)
1025 {
1026  if (this->is_running)
1027  return;
1028 
1029  if (this->system->GetTimestepsTaken() == 0)
1030  {
1031  this->system->SaveStartingPattern();
1032 
1033  // reset the initial number of steps used by system->Update in OnIdle
1034  num_steps = 50;
1035  // 50 is half the initial timesteps_per_render value used in most
1036  // pattern files, but really we could choose any small number > 0
1037  }
1038 
1039  try
1040  {
1041  if (event.GetId() == ID::Step1)
1042  {
1043  this->system->Update(1);
1044  this->pVTKWindow->GetRenderWindow()->GetRenderers()->GetFirstRenderer()->ResetCameraClippingRange();
1045  }
1046  else if (event.GetId() == ID::StepN)
1047  {
1048  // timesteps_per_render might be huge, so don't do this:
1049  // this->system->Update(this->render_settings.GetProperty("timesteps_per_render").GetInt());
1050  // instead we let OnIdle do the stepping, but stop at next render
1051  this->is_running = true;
1052  steps_since_last_render = 0;
1053  this->computation_time_since_last_render = 0.0;
1054  do_one_render = true;
1055  }
1056  }
1057  catch(const exception& e)
1058  {
1059  MonospaceMessageBox(_("An error occurred when running the simulation:\n\n")+wxString(e.what(),wxConvUTF8),_("Error"),wxART_ERROR);
1060  }
1061  catch(...)
1062  {
1063  wxMessageBox(_("An unknown error occurred when running the simulation"));

1193 void MyFrame::OnIdle(wxIdleEvent& event)
1194 {
1195  #ifdef __WXMAC__
1196  // do NOT do this in the Win app (buttons in Info/Help pane won't work)
1197  if (this->IsActive()) this->CheckFocus();
1198  #endif
1199 
1200  // we drive our simulation loop via idle events
1201  if (this->is_running)
1202  {
1203  // ensure num_steps <= timesteps_per_render
1204  int timesteps_per_render = this->render_settings.GetProperty("timesteps_per_render").GetInt();
1205  if (num_steps > timesteps_per_render) num_steps = timesteps_per_render;
1206 
1207  // use temp_steps for the actual system->Update call because it might be < num_steps
1208  int temp_steps = num_steps;
1209  if (steps_since_last_render + temp_steps > timesteps_per_render) {
1210  // do final steps of this rendering phase
1211  temp_steps = timesteps_per_render - steps_since_last_render;
1212  }
1213 
1214  double time_before = get_time_in_seconds();
1215 
1216  try
1217  {
1218  this->system->Update(temp_steps);
1219  this->pVTKWindow->GetRenderWindow()->GetRenderers()->GetFirstRenderer()->ResetCameraClippingRange();
1220  }
1221  catch(const exception& e)
1222  {
1223  this->is_running = false;
1224  this->SetStatusBarText();
1225  this->UpdateToolbars();
1226  MonospaceMessageBox(_("An error occurred when running the simulation:\n\n")+wxString(e.what(),wxConvUTF8),_("Error"),wxART_ERROR);
1227  }
1228  catch(...)
1229  {
1230  this->is_running = false;
1231  this->SetStatusBarText();
1232  this->UpdateToolbars();
1233  wxMessageBox(_("An unknown error occurred when running the simulation"));
1234  }
1235 
1236  double time_diff = get_time_in_seconds() - time_before;
1237 
1238  // note that we don't change num_steps if temp_steps < num_steps
1239  if (num_steps == temp_steps) {
1240  // if the above system->Update was quick then we'll use more steps in the next call,
1241  // otherwise we'll use less steps so that the app remains responsive
1242  if (time_diff < 0.1) {
1243  num_steps *= 2;
1244  if (num_steps > timesteps_per_render) num_steps = timesteps_per_render;
1245  } else {
1246  num_steps /= 2;
1247  if (num_steps < 1) num_steps = 1;
1248  }
1249  }
1250 
1251  this->computation_time_since_last_render += time_diff;
1252  steps_since_last_render += temp_steps;
1253 
1254  if (steps_since_last_render >= timesteps_per_render) {
1255  // it's time to render what we've computed so far
1256  int n_cells = this->system->GetNumberOfCells();
1257  if (this->computation_time_since_last_render == 0.0)
1258  this->computation_time_since_last_render = 0.000001; // unlikely, but play safe
1259  double time_since_last_render = time_before - this->time_at_last_render;
1260  this->time_at_last_render = time_before;
1261  this->timesteps_per_second_buffer[this->i_timesteps_per_second_buffer] = steps_since_last_render / time_since_last_render;
1262  this->computed_frames_per_second_buffer[this->i_timesteps_per_second_buffer] = steps_since_last_render / this->computation_time_since_last_render;
1263  this->i_timesteps_per_second_buffer++;
1264  if(this->i_timesteps_per_second_buffer==10)
1265  {
1266  this->smoothed_timesteps_per_second = 0.0;
1267  double smoothed_cfps = 0.0;
1268  for(int i=0;i<10;i++) {
1269  this->smoothed_timesteps_per_second += this->timesteps_per_second_buffer[i]/10.0;
1270  smoothed_cfps += this->computed_frames_per_second_buffer[i]/10.0;
1271  }
1272  if(smoothed_cfps > this->smoothed_timesteps_per_second)
1273  this->percentage_spent_rendering = 100.0 - 100.0 * this->smoothed_timesteps_per_second / smoothed_cfps;
1274  this->i_timesteps_per_second_buffer = 0;
1275  this->speed_data_available = true;
1276  }
1277 
1278  if(this->is_recording)
1279  this->RecordFrame();
1280 
1281  this->pVTKWindow->Refresh(false);
1282  this->SetStatusBarText();
1283 
1284  if (do_one_render) {
1285  // user selected Step by N so stop now
1286  this->is_running = false;
1287  this->speed_data_available = false;
1288  this->SetStatusBarText();
1289  this->UpdateToolbars();
1290  } else {
1291  // keep simulating
1292  steps_since_last_render = 0;
1293  this->computation_time_since_last_render = 0.0;
1294  }
1295  }
1296 
1297  event.RequestMore(); // trigger another idle event
1298  }
1299 
1300  event.Skip();
1301 }
