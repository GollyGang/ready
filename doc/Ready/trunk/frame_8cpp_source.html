00929         {
00930             wxMessageBox(_("Unsupported format"));

00932         }

01188 
01189 void MyFrame::OnIdle(wxIdleEvent& event)
01190 {
01191     #ifdef __WXMAC__
01192         // do NOT do this in the Win app (buttons in Info/Help pane won't work)
01193         if (this->IsActive()) this->CheckFocus();
01194     #endif
01195     
01196     // we drive our simulation loop via idle events
01197     if (this->is_running)
01198     {
01199         // ensure num_steps <= timesteps_per_render
01200         int timesteps_per_render = this->render_settings.GetProperty("timesteps_per_render").GetInt();
01201         if (num_steps > timesteps_per_render) num_steps = timesteps_per_render;
01202 
01203         // use temp_steps for the actual system->Update call because it might be < num_steps
01204         int temp_steps = num_steps;
01205         if (steps_since_last_render + temp_steps > timesteps_per_render) {
01206             // do final steps of this rendering phase
01207             temp_steps = timesteps_per_render - steps_since_last_render;
01208         }
01209         
01210         double time_before = get_time_in_seconds();
01211         
01212         try 
01213         {
01214             this->system->Update(temp_steps);
01215         }
01216         catch(const exception& e)
01217         {
01218             this->is_running = false;
01219             this->SetStatusBarText();
01220             this->UpdateToolbars();
01221             MonospaceMessageBox(_("An error occurred when running the simulation:\n\n")+wxString(e.what(),wxConvUTF8),_("Error"),wxART_ERROR);
01222         }
01223         catch(...)
01224         {
01225             this->is_running = false;
01226             this->SetStatusBarText();
01227             this->UpdateToolbars();
01228             wxMessageBox(_("An unknown error occurred when running the simulation"));
01229         }
01230         
01231         double time_diff = get_time_in_seconds() - time_before;
01232         
01233         // note that we don't change num_steps if temp_steps < num_steps
01234         if (num_steps == temp_steps) {
01235             // if the above system->Update was quick then we'll use more steps in the next call,
01236             // otherwise we'll use less steps so that the app remains responsive
01237             if (time_diff < 0.1) {
01238                 num_steps *= 2;
01239                 if (num_steps > timesteps_per_render) num_steps = timesteps_per_render;
01240             } else {
01241                 num_steps /= 2;
01242                 if (num_steps < 1) num_steps = 1;
01243             }
01244         }
01245         
01246         this->computation_time_since_last_render += time_diff;
01247         steps_since_last_render += temp_steps;
01248         
01249         if (steps_since_last_render >= timesteps_per_render) {
01250             // it's time to render what we've computed so far
01251             int n_cells = this->system->GetNumberOfCells();
01252             if (this->computation_time_since_last_render == 0.0)
01253                 this->computation_time_since_last_render = 0.000001;  // unlikely, but play safe
01254             double time_since_last_render = time_before - this->time_at_last_render;
01255             this->time_at_last_render = time_before;
01256             this->timesteps_per_second_buffer[this->i_timesteps_per_second_buffer] = steps_since_last_render / time_since_last_render;
01257             this->computed_frames_per_second_buffer[this->i_timesteps_per_second_buffer] = steps_since_last_render / this->computation_time_since_last_render;
01258             this->i_timesteps_per_second_buffer++;
01259             if(this->i_timesteps_per_second_buffer==10)
01260             {
01261                 this->smoothed_timesteps_per_second = 0.0;
01262                 double smoothed_cfps = 0.0;
01263                 for(int i=0;i<10;i++) {
01264                     this->smoothed_timesteps_per_second += this->timesteps_per_second_buffer[i]/10.0;
01265                     smoothed_cfps += this->computed_frames_per_second_buffer[i]/10.0;
01266                 }
01267                 if(smoothed_cfps > this->smoothed_timesteps_per_second)
01268                     this->percentage_spent_rendering = 100.0 - 100.0 * this->smoothed_timesteps_per_second / smoothed_cfps;
01269                 this->i_timesteps_per_second_buffer = 0;
01270                 this->speed_data_available = true;
01271             }
01272 
01273             if(this->is_recording)
01274                 this->RecordFrame();
01275        
01276             this->pVTKWindow->Refresh(false);
01277             this->SetStatusBarText();
01278             
01279             if (do_one_render) {
01280                 // user selected Step by N so stop now
01281                 this->is_running = false;
01282                 this->speed_data_available = false;
01283                 this->SetStatusBarText();
01284                 this->UpdateToolbars();
01285             } else {
01286                 // keep simulating
01287                 steps_since_last_render = 0;
01288                 this->computation_time_since_last_render = 0.0;
01289             }
01290         }
01291         
01292         event.RequestMore(); // trigger another idle event
01293     }
01294     
01295     event.Skip();
01296 }

02182     if (path.StartsWith(readydir)) {
02183         // remove readydir from start of path
02184         path.erase(0, readydir.length());
02185     }
02186 
02187     // duplicate any ampersands so they appear in menu
02188     path.Replace(wxT("&"), wxT("&&"));
02189 
02190     // put given path at start of patternSubMenu
02191     #ifdef __WXGTK__
02192         // avoid wxGTK bug in FindItem if path contains underscores
02193         int id = wxNOT_FOUND;
02194         for (int i = 0; i < numpatterns; i++) {
02195             wxMenuItem* item = patternSubMenu->FindItemByPosition(i);
02196             wxString temp = item->GetText();
02197             temp.Replace(wxT("__"), wxT("_"));
02198             temp.Replace(wxT("&"), wxT("&&"));

02200                 id = ID::OpenRecent + 1 + i;
02201                 break;

02203         }
02204     #else
02205         int id = patternSubMenu->FindItem(path);
02206     #endif

02208         if ( numpatterns < maxpatterns ) {
02209             // add new path
02210             numpatterns++;
02211             id = ID::OpenRecent + numpatterns;

02214             // replace last item with new path
02215             wxMenuItem* item = patternSubMenu->FindItemByPosition(maxpatterns - 1);
02216             item->SetText(path);
02217             id = ID::OpenRecent + maxpatterns;
02218         }
02219     }
02220     
02221     // path exists in patternSubMenu
02222     if ( id > ID::OpenRecent + 1 ) {
02223         // move path to start of menu
02224         wxMenuItem* item;
02225         while ( id > ID::OpenRecent + 1 ) {
02226             wxMenuItem* previtem = patternSubMenu->FindItem(id - 1);

02230                 prevpath.Replace(wxT("__"), wxT("_"));
02231                 prevpath.Replace(wxT("&"), wxT("&&"));
02232             #endif
02233             item = patternSubMenu->FindItem(id);
02234             item->SetText(prevpath);
02235             id--;
02236         }
02237         item = patternSubMenu->FindItem(id);
02238         item->SetText(path);
02239     }
02240

02270 {

02273         wxMenuItem* item = patternSubMenu->FindItemByPosition(pos);

02275         #ifdef __WXGTK__
02276             // remove duplicate underscores

02287             // keep this item

02292             while (nextpos < numpatterns) {

02302                 #endif

02711     int key = event.GetKeyCode();
02712     int mods = event.GetModifiers();
02713 
02714     // WARNING: logic here must match that in KeyComboCtrl::OnChar in prefs.cpp
02715     if (realkey > 0 && mods != wxMOD_NONE) {
02716         #ifdef __WXGTK__
02717             // sigh... wxGTK returns inconsistent results for shift-comma combos
02718             // so we assume that '<' is produced by pressing shift-comma
02719             // (which might only be true for US keyboards)
02720             if (key == '<' && (mods & wxMOD_SHIFT)) realkey = ',';
02721         #endif
02722         #ifdef __WXMSW__
02723             // sigh... wxMSW returns inconsistent results for some shift-key combos
02724             // so again we assume we're using a US keyboard
02725             if (key == '~' && (mods & wxMOD_SHIFT)) realkey = '`';
02726             if (key == '+' && (mods & wxMOD_SHIFT)) realkey = '=';
02727         #endif
02728         if (mods == wxMOD_SHIFT && key != realkey) {
02729             // use translated key code but remove shift key;
02730             // eg. we want shift-'/' to be seen as '?'
02731             mods = wxMOD_NONE;
02732         } else {
02733             // use key code seen by OnKeyDown
02734             key = realkey;
02735             if (key >= 'A' && key <= 'Z') key += 32;  // convert A..Z to a..z
02736         }
02737     }
02738     
02739     #ifdef __WXGTK__
02740         if (realkey == -666) {
02741             // OnKeyDown saw that shift-space was pressed but for some reason
02742             // OnChar doesn't see the modifier (ie. mods is wxMOD_NONE)
02743             key = ' ';
02744             mods = wxMOD_SHIFT;
02745         }
02746     #endif
02747     
02748     if (this->pVTKWindow->HasFocus()) {
02749         ProcessKey(key, mods);
02750         // don't call default handler (wxVTKRenderWindowInteractor::OnChar)
02751         return;
02752     }
02753     
02754     if (this->patterns_panel->TreeHasFocus()) {
02755         // process keyboard shortcut for patterns panel
02756         if (this->patterns_panel->DoKey(key, mods)) return;
02757         // else call default handler
02758         event.Skip();
02759         return;
02760     }
02761     
02762     if (this->info_panel->HtmlHasFocus()) {
02763         // process keyboard shortcut for info panel
02764         if (this->info_panel->DoKey(key, mods)) return;
02765         // else call default handler
02766         event.Skip();
02767         return;
02768     }
02769     
02770     if (this->help_panel->HtmlHasFocus()) {
02771         // process keyboard shortcut for help panel
02772         if (this->help_panel->DoKey(key, mods)) return;
02773         // else call default handler
02774         event.Skip();
02775         return;
02776     }

02853 bool MyFrame::SetDimensions(int x,int y,int z)
02854 {
02855     try 
02856     {
02857         if(x<1 || y<1 || z<1) throw runtime_error("Dimensions must be at least 1");
02858         if( x%this->system->GetBlockSizeX() || y%this->system->GetBlockSizeY() || z%this->system->GetBlockSizeZ() )
02859         {
02860             ostringstream oss;
02861             oss << "Dimensions must be a multiple of the current block size (" << this->system->GetBlockSizeX() << 
02862                 "x" << this->system->GetBlockSizeY() << "x" << this->system->GetBlockSizeZ() << ")";
02863             throw runtime_error(oss.str().c_str());
02864         }
02865         // rearrange the dimensions (for visualization we need the z to be 1 for 2D images, and both y and z to be 1 for 1D images)
02866         if( (x==1 && (y>1 || z>1)) || (y==1 && z>1) )
02867         {
02868             float d[3]={x,y,z};
02869             sort(d,d+3);
02870             if(d[2]!=x || d[1]!=y || d[0]!=z) {
02871                 x=d[2]; y=d[1]; z=d[0];
02872                 wxString msg = _("We've rearranged the order of the dimensions for visualization. New dimensions: ");
02873                 msg << x << _T(" x ") << y << _T(" x ") << z;
02874                 wxMessageBox(msg);
02875             }
02876         }
02877         // attempt the size change
02878         this->system->SetDimensions(x,y,z);
02879     }
02880     catch(const exception& e)
02881     {
02882         MonospaceMessageBox(_("Dimensions not permitted:\n\n")+wxString(e.what(),wxConvUTF8),_("Error"),wxART_ERROR);
02883         return false;
02884     }
02885     catch(...)
02886     {
02887         wxMessageBox(_("Dimensions not permitted"));
02888         return false;
02889     }
02890     this->system->GenerateInitialPattern();
02891     InitializeVTKPipeline(this->pVTKWindow,this->system,this->render_settings,true);
02892     this->UpdateWindows();
02893     return true;
02894 }
