00993 void MyFrame::OnStep(wxCommandEvent& event)
00994 {
00995     if (this->is_running)
00996         return;
00997     
00998     if (this->system->GetTimestepsTaken() == 0)
00999     {
01000         this->system->SaveStartingPattern();
01001     
01002         // reset the initial number of steps used by system->Update in OnIdle
01003         num_steps = 50;
01004         // 50 is half the initial timesteps_per_render value used in most
01005         // pattern files, but really we could choose any small number > 0
01006     }
01007 
01008     try
01009     {
01010         if (event.GetId() == ID::Step1) {
01011             this->system->Update(1);
01012         } else if (event.GetId() == ID::StepN) {
01013             // timesteps_per_render might be huge, so don't do this:
01014             // this->system->Update(this->render_settings.GetProperty("timesteps_per_render").GetInt());
01015             // instead we let OnIdle do the stepping, but stop at next render
01016             this->is_running = true;
01017             steps_since_last_render = 0;
01018             accumulated_time = 0.0;
01019             do_one_render = true;
01020         }
01021     }
01022     catch(const exception& e)
01023     {

01155 
01156 void MyFrame::OnIdle(wxIdleEvent& event)
01157 {
01158     #ifdef __WXMAC__
01159         // do NOT do this in the Win app (buttons in Info/Help pane won't work)
01160         if (this->IsActive()) this->CheckFocus();
01161     #endif
01162     
01163     // we drive our simulation loop via idle events
01164     if (this->is_running)
01165     {
01166         // ensure num_steps <= timesteps_per_render
01167         int timesteps_per_render = this->render_settings.GetProperty("timesteps_per_render").GetInt();
01168         if (num_steps > timesteps_per_render) num_steps = timesteps_per_render;
01169 
01170         // use temp_steps for the actual system->Update call because it might be < num_steps
01171         int temp_steps = num_steps;
01172         if (steps_since_last_render + temp_steps > timesteps_per_render) {
01173             // do final steps of this rendering phase
01174             temp_steps = timesteps_per_render - steps_since_last_render;
01175         }
01176         
01177         double time_before = get_time_in_seconds();
01178         
01179         try 
01180         {
01181             this->system->Update(temp_steps);
01182         }
01183         catch(const exception& e)
01184         {
01185             this->is_running = false;
01186             this->SetStatusBarText();
01187             this->UpdateToolbars();
01188             MonospaceMessageBox(_("An error occurred when running the simulation:\n\n")+wxString(e.what(),wxConvUTF8),_("Error"),wxART_ERROR);
01189         }
01190         catch(...)
01191         {
01192             this->is_running = false;
01193             this->SetStatusBarText();
01194             this->UpdateToolbars();
01195             wxMessageBox(_("An unknown error occurred when running the simulation"));
01196         }
01197         
01198         double time_diff = get_time_in_seconds() - time_before;
01199         
01200         // note that we don't change num_steps if temp_steps < num_steps
01201         if (num_steps == temp_steps) {
01202             // if the above system->Update was quick then we'll use more steps in the next call,
01203             // otherwise we'll use less steps so that the app remains responsive
01204             if (time_diff < 0.1) {
01205                 num_steps *= 2;
01206                 if (num_steps > timesteps_per_render) num_steps = timesteps_per_render;
01207             } else {
01208                 num_steps /= 2;
01209                 if (num_steps < 1) num_steps = 1;
01210             }
01211         }
01212         
01213         accumulated_time += time_diff;
01214         steps_since_last_render += temp_steps;
01215         
01216         if (steps_since_last_render >= timesteps_per_render) {
01217             // it's time to render what we've computed so far
01218             int n_cells = this->system->GetNumberOfCells();
01219             if (accumulated_time == 0.0)
01220                 accumulated_time = 0.000001;  // unlikely, but play safe
01221             this->frames_per_second = steps_since_last_render / accumulated_time;
01222             this->million_cell_generations_per_second = this->frames_per_second * n_cells / 1e6;
01223 
01224             if(this->is_recording)
01225                 this->RecordFrame();
01226        
01227             this->pVTKWindow->Refresh(false);
01228             this->SetStatusBarText();
01229             
01230             if (do_one_render) {
01231                 // user selected Step by N so stop now
01232                 this->is_running = false;
01233                 this->SetStatusBarText();
01234                 this->UpdateToolbars();
01235             } else {
01236                 // keep simulating
01237                 steps_since_last_render = 0;
01238                 accumulated_time = 0.0;
01239             }
01240         }
01241         
01242         event.RequestMore(); // trigger another idle event
01243     }
01244     
01245     event.Skip();
01246 }

02548 void MyFrame::OnChar(wxKeyEvent& event)
02549 {
02550     int key = event.GetKeyCode();
02551     int mods = event.GetModifiers();
02552 
02553     // WARNING: logic here must match that in KeyComboCtrl::OnChar in prefs.cpp
02554     if (realkey > 0 && mods != wxMOD_NONE) {
02555         #ifdef __WXGTK__
02556             // sigh... wxGTK returns inconsistent results for shift-comma combos
02557             // so we assume that '<' is produced by pressing shift-comma
02558             // (which might only be true for US keyboards)
02559             if (key == '<' && (mods & wxMOD_SHIFT)) realkey = ',';
02560         #endif
02561         #ifdef __WXMSW__
02562             // sigh... wxMSW returns inconsistent results for some shift-key combos
02563             // so again we assume we're using a US keyboard
02564             if (key == '~' && (mods & wxMOD_SHIFT)) realkey = '`';
02565             if (key == '+' && (mods & wxMOD_SHIFT)) realkey = '=';
02566         #endif
02567         if (mods == wxMOD_SHIFT && key != realkey) {
02568             // use translated key code but remove shift key;
02569             // eg. we want shift-'/' to be seen as '?'
02570             mods = wxMOD_NONE;
02571         } else {
02572             // use key code seen by OnKeyDown
02573             key = realkey;

02578     #ifdef __WXGTK__
02579         if (realkey == -666) {
02580             // OnKeyDown saw that shift-space was pressed but for some reason
02581             // OnChar doesn't see the modifier (ie. mods is wxMOD_NONE)
02582             key = ' ';
02583             mods = wxMOD_SHIFT;
02584         }
02585     #endif

02692 bool MyFrame::SetDimensions(int x,int y,int z)
02693 {
02694     try 
02695     {
02696         if(x<1 || y<1 || z<1) throw runtime_error("Dimensions must be at least 1");
02697         if( x%this->system->GetBlockSizeX() || y%this->system->GetBlockSizeY() || z%this->system->GetBlockSizeZ() )
02698         {
02699             ostringstream oss;
02700             oss << "Dimensions must be a multiple of the current block size (" << this->system->GetBlockSizeX() << 
02701                 "x" << this->system->GetBlockSizeY() << "x" << this->system->GetBlockSizeZ() << ")";
02702             throw runtime_error(oss.str().c_str());
02703         }
02704         // rearrange the dimensions (for visualization we need the z to be 1 for 2D images, and both y and z to be 1 for 1D images)
02705         if( (x==1 && (y>1 || z>1)) || (y==1 && z>1) )
02706         {
02707             float d[3]={x,y,z};
02708             sort(d,d+3);
02709             if(d[2]!=x || d[1]!=y || d[0]!=z) {
02710                 x=d[2]; y=d[1]; z=d[0];
02711                 wxString msg = _("We've rearranged the order of the dimensions for visualization. New dimensions: ");
02712                 msg << x << _T(" x ") << y << _T(" x ") << z;
02713                 wxMessageBox(msg);
02714             }
02715         }
02716         // attempt the size change
