00899 void ImageRD::GetAsMesh(vtkPolyData *out, const Properties &render_settings) const

00906     float high = render_settings.GetProperty("high").GetFloat();
00907     float vertical_scale_1D = render_settings.GetProperty("vertical_scale_1D").GetFloat();
00908     float vertical_scale_2D = render_settings.GetProperty("vertical_scale_2D").GetFloat();
00909 
00910     switch(this->GetArenaDimensionality())
00911     {
00912         case 1:
00913             {
00914                 float scaling = vertical_scale_1D / (high-low); // vertical_scale gives the height of the graph in worldspace units
00915 
00916                 vtkSmartPointer<vtkImageDataGeometryFilter> plane = vtkSmartPointer<vtkImageDataGeometryFilter>::New();
00917                 plane->SetInput(this->GetImage(iActiveChemical));
00918                 vtkSmartPointer<vtkWarpScalar> warp = vtkSmartPointer<vtkWarpScalar>::New();
00919                 warp->SetInputConnection(plane->GetOutputPort());
00920                 warp->SetScaleFactor(-scaling);
00921                 warp->Update();
00922                 out->DeepCopy(warp->GetOutput());
00923             }
00924             break;
00925         case 2:
00926             {
00927                 float scaling = vertical_scale_2D / (high-low); // vertical_scale gives the height of the graph in worldspace units
00928 
00929                 vtkSmartPointer<vtkImageDataGeometryFilter> plane = vtkSmartPointer<vtkImageDataGeometryFilter>::New();
00930                 plane->SetInput(this->GetImage(iActiveChemical));
00931                 vtkSmartPointer<vtkWarpScalar> warp = vtkSmartPointer<vtkWarpScalar>::New();
00932                 warp->SetInputConnection(plane->GetOutputPort());
00933                 warp->SetScaleFactor(scaling);
00934                 vtkSmartPointer<vtkPolyDataNormals> normals = vtkSmartPointer<vtkPolyDataNormals>::New();
00935                 normals->SetInputConnection(warp->GetOutputPort());
00936                 normals->SplittingOff();
00937                 normals->Update();
00938                 out->DeepCopy(normals->GetOutput());
00939             }
00940             break;
00941         case 3:
00942             if(use_image_interpolation)
00943             {
00944                 // turns the 3d grid of sampled values into a polygon mesh for rendering,
00945                 // by making a surface that contours the volume at a specified level    
00946                 vtkSmartPointer<vtkContourFilter> surface = vtkSmartPointer<vtkContourFilter>::New();
00947                 surface->SetInput(this->GetImage(iActiveChemical));
00948                 surface->SetValue(0, contour_level);
00949                 surface->Update();
00950                 out->DeepCopy(surface->GetOutput());
00951             }
00952             else
00953             {
00954                 // render as cubes, Minecraft-style
00955                 vtkImageData *image = this->GetImage(iActiveChemical);
00956                 int *extent = image->GetExtent();
00957 
00958                 vtkSmartPointer<vtkImageWrapPad> pad = vtkSmartPointer<vtkImageWrapPad>::New();
00959                 pad->SetInput(image);
00960                 pad->SetOutputWholeExtent(extent[0],extent[1]+1,extent[2],extent[3]+1,extent[4],extent[5]+1);
00961                 pad->Update();
00962                 pad->GetOutput()->GetCellData()->SetScalars(image->GetPointData()->GetScalars()); // a non-pipelined operation
00963 
00964                 vtkSmartPointer<vtkThreshold> threshold = vtkSmartPointer<vtkThreshold>::New();
00965                 threshold->SetInputConnection(pad->GetOutputPort());
00966                 threshold->SetInputArrayToProcess(0, 0, 0,
00967                     vtkDataObject::FIELD_ASSOCIATION_CELLS,
00968                     vtkDataSetAttributes::SCALARS);
00969                 threshold->ThresholdByUpper(contour_level);
00970 
00971                 vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
00972                 transform->Translate (-.5, -.5, -.5);
00973                 vtkSmartPointer<vtkTransformFilter> transformModel = vtkSmartPointer<vtkTransformFilter>::New();
00974                 transformModel->SetTransform(transform);
00975                 transformModel->SetInputConnection(threshold->GetOutputPort());
00976 
00977                 vtkSmartPointer<vtkGeometryFilter> geometry = vtkSmartPointer<vtkGeometryFilter>::New();
00978                 geometry->SetInputConnection(transformModel->GetOutputPort());
00979                 geometry->Update();
00980 
00981                 out->DeepCopy(geometry->GetOutput());
00982             }
00983             break;
00984     }
00985 }
