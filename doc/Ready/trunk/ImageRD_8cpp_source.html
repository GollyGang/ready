00872 void ImageRD::GetAsMesh(vtkPolyData *out, const Properties &render_settings) const
00873 {
00874     bool use_image_interpolation = render_settings.GetProperty("use_image_interpolation").GetBool();
00875     int iActiveChemical = IndexFromChemicalName(render_settings.GetProperty("active_chemical").GetChemical());
00876     float contour_level = render_settings.GetProperty("contour_level").GetFloat();
00877 
00878     float low = render_settings.GetProperty("low").GetFloat();
00879     float high = render_settings.GetProperty("high").GetFloat();
00880     float vertical_scale_1D = render_settings.GetProperty("vertical_scale_1D").GetFloat();
00881     float vertical_scale_2D = render_settings.GetProperty("vertical_scale_2D").GetFloat();
00882 
00883     switch(this->GetArenaDimensionality())
00884     {
00885         case 1:
00886             {
00887                 float scaling = vertical_scale_1D / (high-low); // vertical_scale gives the height of the graph in worldspace units
00888 
00889                 vtkSmartPointer<vtkImageDataGeometryFilter> plane = vtkSmartPointer<vtkImageDataGeometryFilter>::New();
00890                 plane->SetInput(this->GetImage(iActiveChemical));
00891                 vtkSmartPointer<vtkWarpScalar> warp = vtkSmartPointer<vtkWarpScalar>::New();
00892                 warp->SetInputConnection(plane->GetOutputPort());
00893                 warp->SetScaleFactor(-scaling);
00894                 warp->Update();
00895                 out->DeepCopy(warp->GetOutput());
00896             }
00897             break;
00898         case 2:
00899             {
00900                 float scaling = vertical_scale_2D / (high-low); // vertical_scale gives the height of the graph in worldspace units
00901 
00902                 vtkSmartPointer<vtkImageDataGeometryFilter> plane = vtkSmartPointer<vtkImageDataGeometryFilter>::New();
00903                 plane->SetInput(this->GetImage(iActiveChemical));
00904                 vtkSmartPointer<vtkWarpScalar> warp = vtkSmartPointer<vtkWarpScalar>::New();
00905                 warp->SetInputConnection(plane->GetOutputPort());
00906                 warp->SetScaleFactor(scaling);
00907                 vtkSmartPointer<vtkPolyDataNormals> normals = vtkSmartPointer<vtkPolyDataNormals>::New();
00908                 normals->SetInputConnection(warp->GetOutputPort());
00909                 normals->SplittingOff();
00910                 normals->Update();
00911                 out->DeepCopy(normals->GetOutput());
00912             }
00913             break;
00914         case 3:
00915             if(use_image_interpolation)
00916             {
00917                 // turns the 3d grid of sampled values into a polygon mesh for rendering,
00918                 // by making a surface that contours the volume at a specified level    
00919                 vtkSmartPointer<vtkContourFilter> surface = vtkSmartPointer<vtkContourFilter>::New();
00920                 surface->SetInput(this->GetImage(iActiveChemical));
00921                 surface->SetValue(0, contour_level);
00922                 surface->Update();
00923                 out->DeepCopy(surface->GetOutput());
00924             }
00925             else
00926             {
00927                 // render as cubes, Minecraft-style
00928                 vtkImageData *image = this->GetImage(iActiveChemical);
00929                 int *extent = image->GetExtent();
00930 
00931                 vtkSmartPointer<vtkImageWrapPad> pad = vtkSmartPointer<vtkImageWrapPad>::New();
00932                 pad->SetInput(image);
00933                 pad->SetOutputWholeExtent(extent[0],extent[1]+1,extent[2],extent[3]+1,extent[4],extent[5]+1);
00934                 pad->Update();
00935                 pad->GetOutput()->GetCellData()->SetScalars(image->GetPointData()->GetScalars()); // a non-pipelined operation
00936 
00937                 vtkSmartPointer<vtkThreshold> threshold = vtkSmartPointer<vtkThreshold>::New();
00938                 threshold->SetInputConnection(pad->GetOutputPort());
00939                 threshold->SetInputArrayToProcess(0, 0, 0,
00940                     vtkDataObject::FIELD_ASSOCIATION_CELLS,
00941                     vtkDataSetAttributes::SCALARS);
00942                 threshold->ThresholdByUpper(contour_level);
00943 
00944                 vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
00945                 transform->Translate (-.5, -.5, -.5);
00946                 vtkSmartPointer<vtkTransformFilter> transformModel = vtkSmartPointer<vtkTransformFilter>::New();
00947                 transformModel->SetTransform(transform);
00948                 transformModel->SetInputConnection(threshold->GetOutputPort());
00949 
00950                 vtkSmartPointer<vtkGeometryFilter> geometry = vtkSmartPointer<vtkGeometryFilter>::New();
00951                 geometry->SetInputConnection(transformModel->GetOutputPort());
00952                 geometry->Update();
00953 
00954                 out->DeepCopy(geometry->GetOutput());
00955             }
00956             break;
00957     }
00958 }
