00891 void ImageRD::GetAsMesh(vtkPolyData *out, const Properties &render_settings) const
00892 {
00893     bool use_image_interpolation = render_settings.GetProperty("use_image_interpolation").GetBool();
00894     int iActiveChemical = IndexFromChemicalName(render_settings.GetProperty("active_chemical").GetChemical());
00895     float contour_level = render_settings.GetProperty("contour_level").GetFloat();
00896 
00897     float low = render_settings.GetProperty("low").GetFloat();
00898     float high = render_settings.GetProperty("high").GetFloat();
00899     float vertical_scale_1D = render_settings.GetProperty("vertical_scale_1D").GetFloat();
00900     float vertical_scale_2D = render_settings.GetProperty("vertical_scale_2D").GetFloat();
00901 
00902     switch(this->GetArenaDimensionality())
00903     {
00904         case 1:
00905             {
00906                 float scaling = vertical_scale_1D / (high-low); // vertical_scale gives the height of the graph in worldspace units
00907 
00908                 vtkSmartPointer<vtkImageDataGeometryFilter> plane = vtkSmartPointer<vtkImageDataGeometryFilter>::New();
00909                 plane->SetInput(this->GetImage(iActiveChemical));
00910                 vtkSmartPointer<vtkWarpScalar> warp = vtkSmartPointer<vtkWarpScalar>::New();
00911                 warp->SetInputConnection(plane->GetOutputPort());
00912                 warp->SetScaleFactor(-scaling);
00913                 warp->Update();
00914                 out->DeepCopy(warp->GetOutput());
00915             }
00916             break;
00917         case 2:
00918             {
00919                 float scaling = vertical_scale_2D / (high-low); // vertical_scale gives the height of the graph in worldspace units
00920 
00921                 vtkSmartPointer<vtkImageDataGeometryFilter> plane = vtkSmartPointer<vtkImageDataGeometryFilter>::New();
00922                 plane->SetInput(this->GetImage(iActiveChemical));
00923                 vtkSmartPointer<vtkWarpScalar> warp = vtkSmartPointer<vtkWarpScalar>::New();
00924                 warp->SetInputConnection(plane->GetOutputPort());
00925                 warp->SetScaleFactor(scaling);
00926                 vtkSmartPointer<vtkPolyDataNormals> normals = vtkSmartPointer<vtkPolyDataNormals>::New();
00927                 normals->SetInputConnection(warp->GetOutputPort());
00928                 normals->SplittingOff();
00929                 normals->Update();
00930                 out->DeepCopy(normals->GetOutput());
00931             }
00932             break;
00933         case 3:
00934             if(use_image_interpolation)
00935             {
00936                 // turns the 3d grid of sampled values into a polygon mesh for rendering,
00937                 // by making a surface that contours the volume at a specified level    
00938                 vtkSmartPointer<vtkContourFilter> surface = vtkSmartPointer<vtkContourFilter>::New();
00939                 surface->SetInput(this->GetImage(iActiveChemical));
00940                 surface->SetValue(0, contour_level);
00941                 surface->Update();
00942                 out->DeepCopy(surface->GetOutput());
00943             }
00944             else
00945             {
00946                 // render as cubes, Minecraft-style
00947                 vtkImageData *image = this->GetImage(iActiveChemical);
00948                 int *extent = image->GetExtent();
00949 
00950                 vtkSmartPointer<vtkImageWrapPad> pad = vtkSmartPointer<vtkImageWrapPad>::New();
00951                 pad->SetInput(image);
00952                 pad->SetOutputWholeExtent(extent[0],extent[1]+1,extent[2],extent[3]+1,extent[4],extent[5]+1);
00953                 pad->Update();
00954                 pad->GetOutput()->GetCellData()->SetScalars(image->GetPointData()->GetScalars()); // a non-pipelined operation
00955 
00956                 vtkSmartPointer<vtkThreshold> threshold = vtkSmartPointer<vtkThreshold>::New();
00957                 threshold->SetInputConnection(pad->GetOutputPort());
00958                 threshold->SetInputArrayToProcess(0, 0, 0,
00959                     vtkDataObject::FIELD_ASSOCIATION_CELLS,
00960                     vtkDataSetAttributes::SCALARS);
00961                 threshold->ThresholdByUpper(contour_level);
00962 
00963                 vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
00964                 transform->Translate (-.5, -.5, -.5);
00965                 vtkSmartPointer<vtkTransformFilter> transformModel = vtkSmartPointer<vtkTransformFilter>::New();
00966                 transformModel->SetTransform(transform);
00967                 transformModel->SetInputConnection(threshold->GetOutputPort());
00968 
00969                 vtkSmartPointer<vtkGeometryFilter> geometry = vtkSmartPointer<vtkGeometryFilter>::New();
00970                 geometry->SetInputConnection(transformModel->GetOutputPort());
00971                 geometry->Update();
00972 
00973                 out->DeepCopy(geometry->GetOutput());
00974             }
00975             break;
00976     }
00977 }
