32 FormulaOpenCLImageRD::FormulaOpenCLImageRD(int opencl_platform,int opencl_device)
33  : OpenCLImageRD(opencl_platform,opencl_device)
34 {

36  this->SetRuleName("Gray-Scott");
37  this->AddParameter("timestep",1.0f);
38  this->AddParameter("D_a",0.082f);

40  this->AddParameter("k",0.06f);

42  this->SetFormula("\

44 delta_b = D_b * laplacian_b + a*b*b - (F+k)*b;");
