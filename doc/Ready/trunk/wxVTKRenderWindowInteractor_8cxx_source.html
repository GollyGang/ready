00867 void wxVTKRenderWindowInteractor::SetRenderWhenDisabled(int newValue)
00868 {
00869   //Change value of __RenderWhenDisabled ivar.
00870   //If __RenderWhenDisabled is false (the default), this widget will not
00871   //call Render() on the RenderWindow if the top level frame (i.e. the
00872   //containing frame) has been disabled.
00873 
00874   //This prevents recursive rendering during wxSafeYield() calls.
00875   //wxSafeYield() can be called during the ProgressMethod() callback of
00876   //a VTK object to have progress bars and other GUI elements updated -
00877   //it does this by disabling all windows (disallowing user-input to
00878   //prevent re-entrancy of code) and then handling all outstanding
00879   //GUI events.
00880         
00881   //However, this often triggers an OnPaint() method for wxVTKRWIs,
00882   //resulting in a Render(), resulting in Update() being called whilst
00883   //still in progress.
00884 
00885   RenderWhenDisabled = (bool)newValue;
00886 }
