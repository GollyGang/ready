26 void Property::ReadFromXML(vtkXMLDataElement* node)
27 {
28  if(this->type=="float")
29  read_required_attribute(node,"value",this->f1);
30  else if(this->type=="int")
31  read_required_attribute(node,"value",this->i);
32  else if(this->type=="bool")
33  {
34  read_required_attribute(node,"value",this->s);
35  if(this->s=="true") this->b = true;
36  else if(this->s=="false") this->b = false;
37  else throw runtime_error("Property::ReadFromXML : unrecognised bool value: "+this->s);
38  }
39  else if(this->type=="color")
40  {
41  read_required_attribute(node,"r",this->f1);
42  read_required_attribute(node,"g",this->f2);
43  read_required_attribute(node,"b",this->f3);
44  }
45  else if(this->type=="chemical")
46  {
47  read_required_attribute(node,"value",this->s);
48  IndexFromChemicalName(this->s);
49  }
50  else if(this->type=="axis")
51  {
52  read_required_attribute(node,"value",this->s);
53  if(this->s != "x" && this->s != "y" && this->s != "z")
54  throw runtime_error("Property::ReadFromXML : unrecognised axis: "+this->s);
55  }
56  else throw runtime_error("Property::ReadFromXML : unrecognised type: "+this->type);
57 }
