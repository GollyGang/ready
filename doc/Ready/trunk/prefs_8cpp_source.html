170 bool ConvertKeyAndModifiers(int wxkey, int wxmods, int* newkey, int* newmods)
171 {
172  // first convert given wx modifiers (set by wxKeyEvent::GetModifiers)
173  // to a corresponding set of mk_* values
174  int ourmods = 0;
175  if (wxmods & wxMOD_CMD) ourmods |= mk_CMD;
176  if (wxmods & wxMOD_ALT) ourmods |= mk_ALT;
177  if (wxmods & wxMOD_SHIFT) ourmods |= mk_SHIFT;
178 #ifdef __WXMAC__
179  if (wxmods & wxMOD_CONTROL) ourmods |= mk_CTRL;
180 #endif
181 
182  // now convert given wx key code to corresponding IK_* code
183  int ourkey;
184  if (wxkey >= 'A' && wxkey <= 'Z') {
185  // convert A..Z to shift+a..shift+z so we can use A..X
186  // for our internal function keys (IK_F1 to IK_F24)
187  ourkey = wxkey + 32;
188  ourmods |= mk_SHIFT;
189 
190  }
191  else if (wxkey >= WXK_F1 && wxkey <= WXK_F24) {
192  // convert wx function key code to IK_F1..IK_F24
193  ourkey = IK_F1 + (wxkey - WXK_F1);
194 
195  }
196  else if (wxkey >= WXK_NUMPAD0 && wxkey <= WXK_NUMPAD9) {
197  // treat numpad digits like ordinary digits
198  ourkey = '0' + (wxkey - WXK_NUMPAD0);
199 
200  }
201  else {
202  switch (wxkey) {
203  case WXK_HOME: ourkey = IK_HOME; break;
204  case WXK_END: ourkey = IK_END; break;
205  case WXK_PAGEUP: ourkey = IK_PAGEUP; break;
206  case WXK_PAGEDOWN: ourkey = IK_PAGEDOWN; break;
207  case WXK_HELP: ourkey = IK_HELP; break;
208  case WXK_INSERT: ourkey = IK_INSERT; break;
209  case WXK_BACK: // treat backspace like delete
210  case WXK_DELETE: ourkey = IK_DELETE; break;
211  case WXK_TAB: ourkey = IK_TAB; break;
212  case WXK_NUMPAD_ENTER: // treat enter like return
213  case WXK_RETURN: ourkey = IK_RETURN; break;
214  case WXK_LEFT: ourkey = IK_LEFT; break;
215  case WXK_RIGHT: ourkey = IK_RIGHT; break;
216  case WXK_UP: ourkey = IK_UP; break;
217  case WXK_DOWN: ourkey = IK_DOWN; break;
218  case WXK_ADD: ourkey = '+'; break;
219  case WXK_SUBTRACT: ourkey = '-'; break;
220  case WXK_DIVIDE: ourkey = '/'; break;
221  case WXK_MULTIPLY: ourkey = '*'; break;
222  default: ourkey = wxkey;
223  }
224  }
225 
226  if (ourkey < 0 || ourkey >= MAX_KEYCODES) return false;
227 
228  *newkey = ourkey;
229  *newmods = ourmods;
230  return true;
231 }

389 void GetKeyAction(char* value)
390 {
391  // parse strings like "z undo" or "space+ctrl advance selection";
392  // note that some errors detected here can be Fatal because the user
393  // has to quit Ready anyway to edit the prefs file
394  char* start = value;
395  char* p = start;
396  int modset = 0;
397  int key = -1;
398 
399  // extract key, skipping first char in case it's '+'
400  if (*p > 0) p++;
401  while (1) {
402  if (*p == 0) {
403  Fatal(wxString::Format(_("Bad key_action value: %s"),
404  wxString(value,wxConvLocal).c_str()));
405  }
406  if (*p == ' ' || *p == '+') {
407  // we found end of key
408  char oldp = *p;
409  *p = 0;
410  int len = strlen(start);
411  if (len == 1) {
412  key = start[0];
413  if (key < ' ' || key > '~') {
414  Fatal(wxString::Format(_("Non-displayable key in key_action: code = %d"), key));
415  }
416  if (key >= 'A' && key <= 'Z') {
417  // convert A..Z to shift+a..shift+z so we can use A..X
418  // for our internal function keys (IK_F1 to IK_F24)
419  key += 32;
420  modset |= mk_SHIFT;
421  }
422  }
423  else if (len > 1) {
424  if ((start[0] == 'f' || start[0] == 'F') && start[1] >= '1' && start[1] <= '9') {
425  // we have a function key
426  char* p = &start[1];
427  int num;
428  sscanf(p, "%d", &num);
429  if (num >= 1 && num <= 24) key = IK_F1 + num - 1;
430  }
431  else {
432  if (ISTRCMP(start, NK_HOME) == 0) key = IK_HOME;
433  else if (ISTRCMP(start, NK_END) == 0) key = IK_END;
434  else if (ISTRCMP(start, NK_PGUP) == 0) key = IK_PAGEUP;
435  else if (ISTRCMP(start, NK_PGDN) == 0) key = IK_PAGEDOWN;
436  else if (ISTRCMP(start, NK_HELP) == 0) key = IK_HELP;
437  else if (ISTRCMP(start, NK_INSERT) == 0) key = IK_INSERT;
438  else if (ISTRCMP(start, NK_DELETE) == 0) key = IK_DELETE;
439  else if (ISTRCMP(start, NK_TAB) == 0) key = IK_TAB;
440  else if (ISTRCMP(start, NK_RETURN) == 0) key = IK_RETURN;
441  else if (ISTRCMP(start, NK_LEFT) == 0) key = IK_LEFT;
442  else if (ISTRCMP(start, NK_RIGHT) == 0) key = IK_RIGHT;
443  else if (ISTRCMP(start, NK_UP) == 0) key = IK_UP;
444  else if (ISTRCMP(start, NK_DOWN) == 0) key = IK_DOWN;
445  else if (ISTRCMP(start, NK_SPACE) == 0) key = ' ';
446  }
447  if (key < 0)
448  Fatal(wxString::Format(_("Unknown key in key_action: %s"),
449  wxString(start,wxConvLocal).c_str()));
450  }
451  *p = oldp; // restore ' ' or '+'
452  start = p;
453  start++;
454  break;
455  }
456  p++;
457  }
458 
459  // *p is ' ' or '+' so extract zero or more modifiers
460  while (*p != ' ') {
461  p++;
462  if (*p == 0) {
463  Fatal(wxString::Format(_("No action in key_action value: %s"),
464  wxString(value,wxConvLocal).c_str()));
465  }
466  if (*p == ' ' || *p == '+') {
467  // we found end of modifier
468  char oldp = *p;
469  *p = 0;
470 #ifdef __WXMAC__
471  if (ISTRCMP(start, "cmd") == 0) modset |= mk_CMD;
472  else if (ISTRCMP(start, "opt") == 0) modset |= mk_ALT;
473  else if (ISTRCMP(start, "ctrl") == 0) modset |= mk_CTRL;
474 #else
475  if (ISTRCMP(start, "ctrl") == 0) modset |= mk_CMD;
476  else if (ISTRCMP(start, "alt") == 0) modset |= mk_ALT;
477 #endif
478  else if (ISTRCMP(start, "shift") == 0) modset |= mk_SHIFT;
479  else
480  Fatal(wxString::Format(_("Unknown modifier in key_action: %s"),
481  wxString(start,wxConvLocal).c_str()));
482  *p = oldp; // restore ' ' or '+'
483  start = p;
484  start++;
485  }
486  }
487 
488  // *p is ' ' so skip and check the action string
489  p++;
490  action_info action = nullaction;
491 
492  // first look for "Open:" followed by file path
493  if (strncmp(p, "Open:", 5) == 0) {
494  action.id = DO_OPENFILE;
495  action.file = wxString(&p[5], wxConvLocal);
496  }
497  else {
498  // assume DO_NOTHING is 0 and start with action 1
499  for (int i = 1; i < MAX_ACTIONS; i++) {
500  if (strcmp(p, GetActionName((action_id) i)) == 0) {
501  action.id = (action_id) i;
502  break;
503  }
504  }
505  }
506 
507  // test for some deprecated actions
508  /*
509  if (action.id == DO_NOTHING) {
510  if (strcmp(p, "Swap Cell Colors") == 0) action.id = DO_INVERT;
511  }
512  */
513 
514  // probably best to silently ignore an unknown action
515  // (friendlier if old Ready is reading newer prefs)
516  /*
517  if (action.id == DO_NOTHING)
518  Warning(wxString::Format(_("Unknown action in key_action: %s"),
519  wxString(p,wxConvLocal).c_str()));
520  */
521 
522  keyaction[key][modset] = action;
523 }

1006 char* linereader::fgets(char* buf, int maxlen)
1007 {
1008  int i = 0;
1009  for (;;) {
1010  if (i+1 >= maxlen) {
1011  buf[i] = 0;
1012  return buf;
1013  }
1014  int c = getc(fp);
1015  switch (c) {
1016  case EOF:
1017  if (i == 0) return 0;
1018  buf[i] = 0;
1019  return buf;
1020  case LF:
1021  if (lastchar != CR) {
1022  lastchar = LF;
1023  buf[i] = 0;
1024  return buf;
1025  }
1026  break;
1027  case CR:
1028  lastchar = CR;
1029  buf[i] = 0;
1030  return buf;
1031  default:
1032  lastchar = c;
1033  buf[i++] = (char)c;
1034  break;
1035  }
1036  }
