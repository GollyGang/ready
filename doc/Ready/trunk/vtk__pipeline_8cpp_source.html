40 void InitializeVTKPipeline(wxVTKRenderWindowInteractor* pVTKWindow,AbstractRD* system,const Properties& render_settings,
41  bool reset_camera)
42 {
43  assert(pVTKWindow);
44  assert(system);
45 
46  // the VTK renderer is responsible for drawing the scene onto the screen
47  vtkSmartPointer<vtkRenderer> pRenderer;
48  if(pVTKWindow->GetRenderWindow()->GetRenderers()->GetNumberOfItems()>0)
49  {
50  pRenderer = pVTKWindow->GetRenderWindow()->GetRenderers()->GetFirstRenderer();
51  pRenderer->RemoveAllViewProps();
52  }
53  else
54  {
55  pRenderer = vtkSmartPointer<vtkRenderer>::New();
56  pVTKWindow->GetRenderWindow()->AddRenderer(pRenderer); // connect it to the window
57 
58  // set the background color
59  pRenderer->GradientBackgroundOn();
60  pRenderer->SetBackground(0,0.4,0.6);
61  pRenderer->SetBackground2(0,0.2,0.3);
62 
63  // change the interactor style to a trackball
64  vtkSmartPointer<vtkInteractorStyleTrackballCamera> is = vtkSmartPointer<vtkInteractorStyleTrackballCamera>::New();
65  pVTKWindow->SetInteractorStyle(is);
66  }
67 
68  system->InitializeRenderPipeline(pRenderer,render_settings);
69 
70  if(reset_camera)
71  {
72  vtkSmartPointer<vtkCamera> camera = vtkSmartPointer<vtkCamera>::New();
73  if(system->GetArenaDimensionality()>2)
74  {
75  camera->SetPosition(-3,3,10); // move off-axis to look more natural
76  }
77  else if( system->GetArenaDimensionality()==2
78  && render_settings.GetProperty("show_displacement_mapped_surface").GetBool()
79  && system->GetFileExtension()!="vtu" )
80  {
81  camera->SetPosition(0,-3,10); // move slightly down, to give perspective on the displacement plots
82  }
83  pRenderer->SetActiveCamera(camera);
84  pRenderer->ResetCamera();
85  }
86 }
