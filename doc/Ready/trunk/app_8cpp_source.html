00042 static void SetAppDirectory(const char* argv0)
00043 {
00044     #if defined(__WXMSW__)
00045         // on Windows we need to reset current directory to app directory if user
00046         // dropped file from somewhere else onto app to start it up (otherwise we
00047         // can't find Help files)
00048         wxString appdir = wxStandardPaths::Get().GetDataDir();
00049         wxString currdir = wxGetCwd();
00050         if ( currdir.CmpNoCase(appdir) != 0 )
00051             wxSetWorkingDirectory(appdir);
00052         // avoid VC++ warning
00053         wxUnusedVar(argv0);
00054     #elif defined(__WXMAC__)
00055         // wxMac has set current directory to location of .app bundle so no need
00056         // to do anything
00057     #elif defined(__WXGTK__)
00058         // first, try to switch to READYDIR if that is set to a sensible value:
00059         static const char *gd = STR(READYDIR);
00060         if ( *gd == '/' && wxSetWorkingDirectory(wxString(gd,wxConvLocal)) ) {
00061             return;
00062         }
00063         // otherwise, use the executable directory as the application directory.
00064         // user might have started app from a different directory so find
00065         // last "/" in argv0 and change cwd if "/" isn't part of "./" prefix
00066         unsigned int pos = strlen(argv0);
00067         while (pos > 0) {
00068             pos--;
00069             if (argv0[pos] == '/') break;
00070         }
00071         if ( pos > 0 && !(pos == 1 && argv0[0] == '.') ) {
00072             char appdir[2048];
00073             if (pos < sizeof(appdir)) {
00074                 strncpy(appdir, argv0, pos);
00075                 appdir[pos] = 0;
00076                 wxSetWorkingDirectory(wxString(appdir,wxConvLocal));
00077             }
00078         }
00079     #endif
00080 }

00102     if (initdir.Last() != wxFILE_SEP_PATH) initdir += wxFILE_SEP_PATH;
00103     
00104     // make sure current working directory contains application otherwise

00119     // argc is > 1 if command line has one or more pattern files
