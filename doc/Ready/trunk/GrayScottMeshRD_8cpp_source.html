00066         else
00067         {
00068             source_a = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(0).c_str()) );

00072         }

00075             // compute the laplacian
00076             aval = source_a->GetValue(iCell);
00077             bval = source_b->GetValue(iCell);
00078             dda = 0.0f;
00079             ddb = 0.0f;
00080             for(int iNeighbor=0;iNeighbor<this->max_neighbors;iNeighbor++)
00081             {
00082                 int k = iCell*this->max_neighbors + iNeighbor;
00083                 neighbor_index = this->cell_neighbor_indices[k];
00084                 diffusion_coefficient = this->cell_neighbor_weights[k];
00085                 dda += source_a->GetValue(neighbor_index) * diffusion_coefficient;
00086                 ddb += source_b->GetValue(neighbor_index) * diffusion_coefficient;
00087             }
00088             dda -= aval;
00089             ddb -= bval;
00090             dda *= 4.0f; // scale the Laplacian to be more similar to the 2D square grid version, so the same parameters work
00091             ddb *= 4.0f;
00092             // avoid denormals
00093             dda += 1e-6f;
00094             ddb += 1e-6f;
00095             // Gray-Scott update step:
00096             da = D_a * dda - aval*bval*bval + F*(1-aval);
00097             db = D_b * ddb + aval*bval*bval - (F+k)*bval;
00098             // apply the step:
00099             target_a->SetValue(iCell,aval + timestep*da );
00100             target_b->SetValue(iCell,bval + timestep*db );
00101         }
00102     }
00103     if(n_steps%2)
00104         this->mesh->DeepCopy(this->buffer);
00105 }
