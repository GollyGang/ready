51 void GrayScottMeshRD::InternalUpdate(int n_steps)
52 {
53  float timestep = this->GetParameterValueByName("timestep");
54  float D_a = this->GetParameterValueByName("D_a");
55  float D_b = this->GetParameterValueByName("D_b");
56  float k = this->GetParameterValueByName("k");
57  float F = this->GetParameterValueByName("F");
58 
59  vtkFloatArray *source_a,*source_b;
60  vtkFloatArray *target_a,*target_b;
61  float dda,ddb,aval,bval,da,db;
62  int neighbor_index;
63  float diffusion_coefficient;
64 
65  for(int iStep=0;iStep<n_steps;iStep++)
66  {
67  if(iStep%2)
68  {
69  source_a = vtkFloatArray::SafeDownCast( this->buffer->GetCellData()->GetArray(GetChemicalName(0).c_str()) );
70  source_b = vtkFloatArray::SafeDownCast( this->buffer->GetCellData()->GetArray(GetChemicalName(1).c_str()) );
71  target_a = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(0).c_str()) );
72  target_b = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(1).c_str()) );
73  }
74  else
75  {
76  source_a = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(0).c_str()) );
77  source_b = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(1).c_str()) );
78  target_a = vtkFloatArray::SafeDownCast( this->buffer->GetCellData()->GetArray(GetChemicalName(0).c_str()) );
79  target_b = vtkFloatArray::SafeDownCast( this->buffer->GetCellData()->GetArray(GetChemicalName(1).c_str()) );
80  }
81  for(vtkIdType iCell=0;iCell<this->mesh->GetNumberOfCells();iCell++)
82  {
83  // compute the laplacian
84  aval = source_a->GetValue(iCell);
85  bval = source_b->GetValue(iCell);
86  dda = 0.0f;
87  ddb = 0.0f;
88  for(int iNeighbor=0;iNeighbor<this->max_neighbors;iNeighbor++)
89  {
90  int k = iCell*this->max_neighbors + iNeighbor;
91  neighbor_index = this->cell_neighbor_indices[k];
92  diffusion_coefficient = this->cell_neighbor_weights[k];
93  dda += source_a->GetValue(neighbor_index) * diffusion_coefficient;
94  ddb += source_b->GetValue(neighbor_index) * diffusion_coefficient;
95  }
96  dda -= aval;
97  ddb -= bval;
98  dda *= 4.0f; // scale the Laplacian to be more similar to the 2D square grid version, so the same parameters work
99  ddb *= 4.0f;
100  // Gray-Scott update step:
101  da = D_a * dda - aval*bval*bval + F*(1-aval);
102  db = D_b * ddb + aval*bval*bval - (F+k)*bval;
103  #if !defined( USE_SSE )
104  // avoid denormals manually
105  da += 1e-10f;
106  db += 1e-10f;
107  #endif
108  // apply the step:
109  target_a->SetValue(iCell,aval + timestep*da );
110  target_b->SetValue(iCell,bval + timestep*db );
111  }
112  }
113  if(n_steps%2)
114  this->mesh->DeepCopy(this->buffer);
115 }
