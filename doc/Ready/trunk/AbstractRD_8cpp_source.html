200 void AbstractRD::InitializeFromXML(vtkXMLDataElement* rd,bool& warn_to_update)
201 {
202  string str;
203  const char *s;
204  float f;
205  int i;
206 
207  // check whether we should warn the user that they need to update Ready
208  {
209  read_required_attribute(rd,"format_version",i);
210  warn_to_update = (i>4);
211  // (we will still proceed and try to read the file but it might fail or give poor results)
212  }
213 
214  vtkSmartPointer<vtkXMLDataElement> rule = rd->FindNestedElementWithName("rule");
215  if(!rule) throw runtime_error("rule node not found in file");
216 
217  // rule_name:
218  read_required_attribute(rule,"name",str);
219  this->SetRuleName(str);
220 
221  // wrap-around
222  s = rule->GetAttribute("wrap");
223  if(!s) this->wrap = true;
224  else this->wrap = (string(s)=="1");
225 
226  // neighborhood specifiers
227 
228  s = rule->GetAttribute("neighborhood_type");
229  if(!s) this->neighborhood_type = VERTEX_NEIGHBORS;
230  else if(this->recognized_neighborhood_type_identifiers.find(s)==this->recognized_neighborhood_type_identifiers.end())
231  throw runtime_error("Unrecognized neighborhood_type");
232  else this->neighborhood_type = this->recognized_neighborhood_type_identifiers[s];
233 
234  s = rule->GetAttribute("neighborhood_range");
235  if(!s) this->neighborhood_range = 1;
236  else {
237  istringstream iss(s);
238  iss >> this->neighborhood_range;
239  }
240  if(neighborhood_range!=1)
241  throw runtime_error("Unsupported neighborhood range");
242 
243  s = rule->GetAttribute("neighborhood_weight");
244  if(!s) this->neighborhood_weight_type = LAPLACIAN;
245  else if(this->recognized_neighborhood_weight_identifiers.find(s)==this->recognized_neighborhood_weight_identifiers.end())
246  throw runtime_error("Unrecognized neighborhood_weight");
247  else this->neighborhood_weight_type = this->recognized_neighborhood_weight_identifiers[s];
248 
249  // parameters:
250  this->DeleteAllParameters();
251  for(int i=0;i<rule->GetNumberOfNestedElements();i++)
252  {
253  vtkSmartPointer<vtkXMLDataElement> node = rule->GetNestedElement(i);
254  if(string(node->GetName())!="param") continue;
255  string name;
256  s = node->GetAttribute("name");
257  if(!s) throw runtime_error("Failed to read param attribute: name");
258  name = trim_multiline_string(s);
259  s = node->GetCharacterData();
260  if(!s || !from_string(s,f)) throw runtime_error("Failed to read param value");
261  this->AddParameter(name,f);
262  }
263 
264  // description:
265  vtkSmartPointer<vtkXMLDataElement> xml_description = rd->FindNestedElementWithName("description");
266  if(!xml_description) this->SetDescription(""); // optional, default is empty string
267  else this->SetDescription(trim_multiline_string(xml_description->GetCharacterData()));
268 
269  // initial_pattern_generator:
270  this->ReadInitialPatternGeneratorFromXML(rd->FindNestedElementWithName("initial_pattern_generator"));
271 }

410 void AbstractRD::Undo()
411 {
412  if(!this->CanUndo()) throw runtime_error("AbstractRD::Undo() : attempt to undo when undo not possible");
413 
414  // find the last done paint action, undo backwards from there until the previous last_of_group (exclusive)
415  vector<PaintAction>::reverse_iterator rit;
416  for(rit=this->undo_stack.rbegin();!rit->done;rit++); // skip over actions that have already been undone
417  while(true)
418  {
419  this->FlipPaintAction(*rit);
420  rit++;
421  if(rit==this->undo_stack.rend() || rit->last_of_group)
422  break;
423  }
424 }

428 void AbstractRD::Redo()
429 {
430  if(!this->CanRedo()) throw runtime_error("AbstractRD::Redo() : attempt to redo when redo not possible");
431 
432  // find the first undone paint action, redo forwards from there until the next last_of_group (inclusive)
433  vector<PaintAction>::iterator it;
434  for(it=this->undo_stack.begin();it->done;it++); // skip over actions that have already been done
435  do
436  {
437  this->FlipPaintAction(*it);
438  if(it->last_of_group)
439  break;
440  it++;
441  } while(it!=this->undo_stack.end());
442 }

446 void AbstractRD::StorePaintAction(int iChemical,int iCell,float old_val)
447 {
448  // forget all stored undone actions
449  while(!this->undo_stack.empty() && !this->undo_stack.back().done)
450  this->undo_stack.pop_back();
451  // add the new paint action
452  PaintAction pa;
453  pa.iCell = iCell;
454  pa.iChemical = iChemical;
455  pa.val = old_val; // (the cell itself stores the new val, we just need the old one)
456  pa.done = true;
457  pa.last_of_group = false;
458  this->undo_stack.push_back(pa);
459 }
