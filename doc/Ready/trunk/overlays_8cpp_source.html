00091 Overlay::Overlay(vtkXMLDataElement* node) : XML_Object(node), op(NULL), fill(NULL)
00092 {
00093     string s;
00094     read_required_attribute(node,"chemical",s);
00095     this->iTargetChemical = IndexFromChemicalName(s);
00096     const int n_nested = node->GetNumberOfNestedElements();
00097     if(n_nested<3)
00098         throw runtime_error("overlay : expected at least 3 nested elements (operation, fill, shape)");
00099     bool parsedOK;
00100     for(int i_nested=0;i_nested<n_nested;i_nested++)
00101     {
00102         vtkXMLDataElement *subnode = node->GetNestedElement(i_nested);
00103         // is this an operation element?
00104         try { this->op = BaseOperation::New(subnode); parsedOK = true; }
00105         catch(runtime_error&) { parsedOK = false; }
00106         if(parsedOK) continue;
00107         // is this a fill element?
00108         try { this->fill = BaseFill::New(subnode); parsedOK = true; }
00109         catch(runtime_error&) { parsedOK = false; }
00110         if(parsedOK) continue;
00111         // must be a shape element
00112         this->shapes.push_back(BaseShape::New(subnode));
00113         // (The usual advice is to not use exceptions for things that are expected to happen but arguably this
00114         //  is an example of where the alternative (return values) is less desirable.)
00115     }
00116     if(this->op == NULL) throw runtime_error("overlay: missing operation element");
00117     if(this->fill == NULL) throw runtime_error("overlay: missing fill element");
00118     if(this->shapes.empty()) throw runtime_error("overlay: missing shape element");
00119 }
