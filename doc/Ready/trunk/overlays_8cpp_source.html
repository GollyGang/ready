91 Overlay::Overlay(vtkXMLDataElement* node) : XML_Object(node), op(NULL), fill(NULL)
92 {
93  string s;
94  read_required_attribute(node,"chemical",s);
95  this->iTargetChemical = IndexFromChemicalName(s);
96  const int n_nested = node->GetNumberOfNestedElements();
97  if(n_nested<3)
98  throw runtime_error("overlay : expected at least 3 nested elements (operation, fill, shape)");
99  BaseShape *pShape;
100  BaseOperation *pOp;
101  BaseFill *pFill;
102  for(int i_nested=0;i_nested<n_nested;i_nested++)
103  {
104  vtkXMLDataElement *subnode = node->GetNestedElement(i_nested);
105  // is this an operation element?
106  pOp = BaseOperation::New(subnode);
107  if(pOp) {
108  this->op = pOp; // TODO: check if already supplied
109  continue; // (save time parsing as other types)
110  }
111  // is this a fill element?
112  pFill = BaseFill::New(subnode);
113  if(pFill) {
114  this->fill = pFill; // TODO: check if already supplied
115  continue;
116  }
117  // must be a shape element?
118  pShape = BaseShape::New(subnode);
119  if(pShape)
120  this->shapes.push_back(pShape);
121  else throw runtime_error(string("Unknown overlay element: ")+subnode->GetName());
122  }
123  if(this->op == NULL) throw runtime_error("overlay: missing operation element");
124  if(this->fill == NULL) throw runtime_error("overlay: missing fill element");
125  if(this->shapes.empty()) throw runtime_error("overlay: missing shape element");
126 }

148 float Overlay::Apply(vector<float> vals,AbstractRD* system,float x,float y,float z) const
149 {
150  float val = vals[this->iTargetChemical];
151  for(int iShape=0;iShape<(int)this->shapes.size();iShape++)
152  {
153  if( this->shapes[iShape]->IsInside( x, y, z, system->GetX(), system->GetY(), system->GetZ(), system->GetArenaDimensionality() ) )
154  {
155  this->op->Apply( val, this->fill->GetValue(system,vals,x,y,z) );
156  vals[this->iTargetChemical] = val; // in case there are multiple shapes at this location in this overlay
157  }
158  }
159  return val;
160 }

446  // project this point onto the linear gradient axis
