34 bool OpenCL_utils::IsOpenCLAvailable()
35 {
36  if(LinkOpenCL()!= CL_SUCCESS)
37  return false; // no OpenCL SDK or driver installed
38 
39  // get available OpenCL platforms
40  const size_t MAX_PLATFORMS = 10;
41  cl_platform_id platforms_available[MAX_PLATFORMS];
42  cl_uint num_platforms;
43  cl_int ret = clGetPlatformIDs(MAX_PLATFORMS,platforms_available,&num_platforms);
44  if(ret != CL_SUCCESS || num_platforms==0)
45  return false; // OpenCL is installed but no platforms available (in a virtualized OS?)
46 
47  // look for OpenCL devices on any platform
48  const size_t MAX_DEVICES = 10;
49  cl_device_id devices_available[MAX_DEVICES];
50  cl_uint num_devices;
51  for(unsigned int iPlatform=0;iPlatform<num_platforms;iPlatform++)
52  {
53  clGetDeviceIDs(platforms_available[iPlatform],CL_DEVICE_TYPE_ALL,MAX_DEVICES,
54  devices_available,&num_devices);
55  if(num_devices>0)
56  return true;
57  }
58  return false; // platforms present but no OpenCL devices available
59 }

89  if(ret != CL_SUCCESS || num_platforms==0)

93  }
94  else
95  {
96  report << "Found " << num_platforms << " platform(s):\n";
97 
98  for(unsigned int iPlatform=0;iPlatform<num_platforms;iPlatform++)
99  {
100  report << "Platform " << iPlatform+1 << ":\n";
101  const size_t MAX_INFO_LENGTH = 1000;
102  char info[MAX_INFO_LENGTH];
103  size_t info_length;
104  for(cl_platform_info i=CL_PLATFORM_PROFILE;i<=CL_PLATFORM_EXTENSIONS;i++)
105  {
106  clGetPlatformInfo(platforms_available[iPlatform],i,MAX_INFO_LENGTH,info,&info_length);
107  report << GetPlatformInfoIdAsString(i) << " : " << info << "\n";
108  }
109 
110  const size_t MAX_DEVICES = 10;
111  cl_device_id devices_available[MAX_DEVICES];
112  cl_uint num_devices;
113  clGetDeviceIDs(platforms_available[iPlatform],CL_DEVICE_TYPE_ALL,MAX_DEVICES,devices_available,&num_devices);
114  report << "\nFound " << num_devices << " device(s) on this platform.\n";
115 
116  const int N_CHAR_RETURNING_IDS = 6;
117  cl_device_info charReturningIds[N_CHAR_RETURNING_IDS] = {CL_DEVICE_EXTENSIONS,CL_DEVICE_NAME,CL_DEVICE_PROFILE,CL_DEVICE_VENDOR,
118  CL_DEVICE_VERSION,CL_DRIVER_VERSION};
119  const int N_UINT_RETURNING_IDS = 13;
120  cl_device_info uintReturningIds[N_UINT_RETURNING_IDS] = {CL_DEVICE_ADDRESS_BITS,CL_DEVICE_GLOBAL_MEM_CACHELINE_SIZE,CL_DEVICE_MAX_CLOCK_FREQUENCY,
121  CL_DEVICE_MAX_COMPUTE_UNITS,CL_DEVICE_MAX_CONSTANT_ARGS,CL_DEVICE_MAX_READ_IMAGE_ARGS,CL_DEVICE_MAX_SAMPLERS,CL_DEVICE_MAX_WORK_ITEM_DIMENSIONS,
122  CL_DEVICE_MAX_WRITE_IMAGE_ARGS,CL_DEVICE_MEM_BASE_ADDR_ALIGN,CL_DEVICE_MIN_DATA_TYPE_ALIGN_SIZE,CL_DEVICE_PREFERRED_VECTOR_WIDTH_FLOAT,
123  CL_DEVICE_VENDOR_ID};
124  const int N_BOOL_RETURNING_IDS = 5;
125  cl_device_info boolReturningIds[N_BOOL_RETURNING_IDS] = {CL_DEVICE_AVAILABLE,CL_DEVICE_COMPILER_AVAILABLE,CL_DEVICE_ENDIAN_LITTLE,CL_DEVICE_ERROR_CORRECTION_SUPPORT,
126  CL_DEVICE_IMAGE_SUPPORT};
127  const int N_ULONG_RETURNING_IDS = 5;
128  cl_device_info ulongReturningIds[N_ULONG_RETURNING_IDS] = {CL_DEVICE_GLOBAL_MEM_CACHE_SIZE,CL_DEVICE_GLOBAL_MEM_SIZE,CL_DEVICE_LOCAL_MEM_SIZE,CL_DEVICE_MAX_CONSTANT_BUFFER_SIZE,
129  CL_DEVICE_MAX_MEM_ALLOC_SIZE};
130  const int N_SIZET_RETURNING_IDS = 8;
131  cl_device_info sizetReturningIds[N_SIZET_RETURNING_IDS] = {CL_DEVICE_IMAGE2D_MAX_HEIGHT,CL_DEVICE_IMAGE2D_MAX_WIDTH,CL_DEVICE_IMAGE3D_MAX_DEPTH,CL_DEVICE_IMAGE3D_MAX_HEIGHT,
132  CL_DEVICE_IMAGE3D_MAX_WIDTH,CL_DEVICE_MAX_PARAMETER_SIZE,CL_DEVICE_MAX_WORK_GROUP_SIZE,CL_DEVICE_PROFILING_TIMER_RESOLUTION};
133 
134  cl_uint uint_value;
135  cl_bool bool_value;
136  cl_ulong ulong_value;
137  size_t sizet_value;
138  for(unsigned int iDevice=0;iDevice<num_devices;iDevice++)
139  {
140  report << "Device " << iDevice+1 << ":\n";
141  for(int i=0;i<N_CHAR_RETURNING_IDS;i++)
142  {
143  clGetDeviceInfo(devices_available[iDevice],charReturningIds[i],MAX_INFO_LENGTH,info,&info_length);
144  report << GetDeviceInfoIdAsString(charReturningIds[i]) << " : " << info << "\n";
145  }
146  for(int i=0;i<N_UINT_RETURNING_IDS;i++)
147  {
148  clGetDeviceInfo(devices_available[iDevice],uintReturningIds[i],sizeof(uint_value),&uint_value,&info_length);
149  report << GetDeviceInfoIdAsString(uintReturningIds[i]) << " : " << uint_value << "\n";
150  }
151  for(int i=0;i<N_BOOL_RETURNING_IDS;i++)
152  {
153  clGetDeviceInfo(devices_available[iDevice],boolReturningIds[i],sizeof(bool_value),&bool_value,&info_length);
154  report << GetDeviceInfoIdAsString(boolReturningIds[i]) << " : " << (bool_value?"yes":"no") << "\n";
155  }
156  for(int i=0;i<N_ULONG_RETURNING_IDS;i++)
157  {
158  clGetDeviceInfo(devices_available[iDevice],ulongReturningIds[i],sizeof(ulong_value),&ulong_value,&info_length);
159  report << GetDeviceInfoIdAsString(ulongReturningIds[i]) << " : " << ulong_value << "\n";
160  }
161  for(int i=0;i<N_SIZET_RETURNING_IDS;i++)
162  {
163  clGetDeviceInfo(devices_available[iDevice],sizetReturningIds[i],sizeof(sizet_value),&sizet_value,&info_length);
164  report << GetDeviceInfoIdAsString(sizetReturningIds[i]) << " : " << sizet_value << "\n";
165  }
166  // CL_DEVICE_MAX_WORK_ITEM_SIZES:
167  size_t dim[3];
168  clGetDeviceInfo(devices_available[iDevice],CL_DEVICE_MAX_WORK_ITEM_SIZES,sizeof(dim),dim,&info_length);
169  report << GetDeviceInfoIdAsString(CL_DEVICE_MAX_WORK_ITEM_SIZES) << " : " << dim[0] << ", " << dim[1] << ", " << dim[2] << "\n";
170  // CL_DEVICE_TYPE:
171  cl_device_type device_type;
172  clGetDeviceInfo(devices_available[iDevice],CL_DEVICE_TYPE,sizeof(device_type),&device_type,&info_length);
173  ostringstream oss;
174  if(device_type&CL_DEVICE_TYPE_CPU) oss << "CPU";
175  if(device_type&CL_DEVICE_TYPE_GPU) { if(!oss.str().empty()) { oss << " & "; } oss << "GPU"; }
176  if(device_type&CL_DEVICE_TYPE_ACCELERATOR) { if(!oss.str().empty()) { oss << " & "; } oss << "ACCELERATOR"; }
177  if(device_type&CL_DEVICE_TYPE_DEFAULT) { if(!oss.str().empty()) { oss << " & "; } oss << "DEFAULT"; }
178  report << GetDeviceInfoIdAsString(CL_DEVICE_TYPE) << " : " << oss.str() << "\n";
179  }
180  report << "\n";
181  }
182  }
183 
184  // bonus feature: report CPU capabilities
185  // TODO: use a cross-platform replacement for __cpuid()
186  #if (defined(_WIN32) || defined(_WIN64))
187  {
188  // http://stackoverflow.com/questions/6121792/is-this-code-valid-to-check-for-sse3
189  int x64 = false;
190  int MMX = false;
191  int SSE = false;
192  int SSE2 = false;
193  int SSE3 = false;
194  int SSSE3 = false;
195  int SSE41 = false;
196  int SSE42 = false;
197  int SSE4a = false;
198  int AVX = false;
199  int XOP = false;
200  int FMA3 = false;
201  int FMA4 = false;
202 
203  int info[4];
204  __cpuid(info, 0);
205  int nIds = info[0];
206 
207  __cpuid(info, 0x80000000);
208  int nExIds = info[0];
209 
210  // Detect Instruction Set
211  if (nIds >= 1){
212  __cpuid(info,0x00000001);
213  MMX = (info[3] & ((int)1 << 23)) != 0;
214  SSE = (info[3] & ((int)1 << 25)) != 0;
215  SSE2 = (info[3] & ((int)1 << 26)) != 0;
216  SSE3 = (info[2] & ((int)1 << 0)) != 0;
217 
218  SSSE3 = (info[2] & ((int)1 << 9)) != 0;
219  SSE41 = (info[2] & ((int)1 << 19)) != 0;
220  SSE42 = (info[2] & ((int)1 << 20)) != 0;
221 
222  AVX = (info[2] & ((int)1 << 28)) != 0;
223  FMA3 = (info[2] & ((int)1 << 12)) != 0;
224  }
225 
226  if (nExIds >= 0x80000001){
227  __cpuid(info,0x80000001);
228  x64 = (info[3] & ((int)1 << 29)) != 0;
229  SSE4a = (info[2] & ((int)1 << 6)) != 0;
230  FMA4 = (info[2] & ((int)1 << 16)) != 0;
231  XOP = (info[2] & ((int)1 << 11)) != 0;
232  }
233 
234  report << "----------- CPU information: ------------\n";
235  report << "x64: " << x64 << "\n";
236  report << "MMX: " << MMX << "\n";
237  report << "SSE: " << SSE << "\n";
238  report << "SSE2: " << SSE2 << "\n";
239  report << "SSE3: " << SSE3 << "\n";
240  report << "SSSE3: " << SSSE3 << "\n";
241  report << "SSE41: " << SSE41 << "\n";
242  report << "SSE42: " << SSE42 << "\n";
243  report << "SSE4a: " << SSE4a << "\n";
244  report << "AVX: " << AVX << "\n";
245  report << "XOP: " << XOP << "\n";
246  report << "FMA3: " << FMA3 << "\n";
247  report << "FMA4: " << FMA4 << "\n";
248  }
249  #endif
250 
251  return report.str();
252 }
