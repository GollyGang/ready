00034 bool OpenCL_utils::IsOpenCLAvailable()
00035 {
00036     if(LinkOpenCL()!= CL_SUCCESS)
00037                 return false; // no OpenCL SDK or driver installed
00038 
00039     // get available OpenCL platforms
00040     const size_t MAX_PLATFORMS = 10;
00041     cl_platform_id platforms_available[MAX_PLATFORMS];
00042     cl_uint num_platforms;
00043     cl_int ret = clGetPlatformIDs(MAX_PLATFORMS,platforms_available,&num_platforms);
00044     if(ret != CL_SUCCESS || num_platforms==0)
00045                 return false; // OpenCL is installed but no platforms available (in a virtualized OS?)
00046 
00047         // look for OpenCL devices on any platform
00048         const size_t MAX_DEVICES = 10;
00049         cl_device_id devices_available[MAX_DEVICES];
00050         cl_uint num_devices;
00051     for(unsigned int iPlatform=0;iPlatform<num_platforms;iPlatform++)
00052     {
00053         clGetDeviceIDs(platforms_available[iPlatform],CL_DEVICE_TYPE_ALL,MAX_DEVICES,
00054                         devices_available,&num_devices);
00055                 if(num_devices>0)
00056                         return true;
00057         }
00058         return false; // platforms present but no OpenCL devices available
00059 }
