00089 void HtmlView::OnLinkClicked(const wxHtmlLinkInfo& link)
00090 {
00091     wxString url = link.GetHref();
00092     if ( url.StartsWith(wxT("http:")) || url.StartsWith(wxT("mailto:")) ) {
00093     // pass http/mailto URL to user's preferred browser/emailer
00094     #if defined(__WXMAC__)
00095         // wxLaunchDefaultBrowser doesn't work on Mac with IE (get msg in console.log)
00096         // but it's easier just to use the Mac OS X open command
00097         if ( wxExecute(wxT("open ") + url, wxEXEC_ASYNC) == -1 )
00098             Warning(_("Could not open URL!"));
00099     #elif defined(__WXGTK__)
00100         // wxLaunchDefaultBrowser is not reliable on Linux/GTK so we call gnome-open
00101         if ( wxExecute(wxT("gnome-open ") + url, wxEXEC_ASYNC) == -1 )
00102             Warning(_("Could not open URL!"));
00103     #else
00104         if ( !wxLaunchDefaultBrowser(url) )
00105             Warning(_("Could not launch browser!"));
00106     #endif
00107 
00108     } else if ( url.StartsWith(_T("prefs:")) ) {
00109         // user clicked on link to Preferences dialog
00110         frame->ShowPrefsDialog( url.AfterFirst(':') );
00111 
00112     } else if ( url.StartsWith(wxT("edit:")) ) {
00113         // open clicked file in user's preferred text editor
00114         wxString path = url.AfterFirst(':');
00115         #ifdef __WXMSW__
00116             path.Replace(wxT("/"), wxT("\\"));
00117         #endif
00118         wxFileName fname(path);
00119         if (!fname.IsAbsolute()) path = readydir + path;
00120         frame->EditFile(path);
00121 
00122 
00123     } else if ( url.StartsWith(wxT("open:")) ) {
00124         // open clicked file
00125         wxString path = url.AfterFirst(':');
00126         #ifdef __WXMSW__
00127             path.Replace(wxT("/"), wxT("\\"));
00128         #endif
00129         wxFileName fname(path);
00130         if (!fname.IsAbsolute()) path = readydir + path;
00131         if (editlink) {
00132             frame->EditFile(path);
00133         } else {
00134             frame->Raise();
00135             frame->OpenFile(path);
00136         }
00137 
00138     } else {
00139         // assume it's a link to a local target or another help file
00140         panel->ShowHelp(url);
00141     }
