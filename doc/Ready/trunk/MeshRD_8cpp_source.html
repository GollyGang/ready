148 void MeshRD::GenerateInitialPattern()
149 {
150  this->BlankImage();
151 
152  vtkIdType npts,*pts;
153  float cp[3];
154  vtkFloatingPointType *bounds = this->mesh->GetBounds();
155  for(vtkIdType iCell=0;iCell<this->mesh->GetNumberOfCells();iCell++)
156  {
157  this->mesh->GetCellPoints(iCell,npts,pts);
158  // get a point at the centre of the cell (need a location to sample the overlays)
159  cp[0]=cp[1]=cp[2]=0.0f;
160  for(vtkIdType iPt=0;iPt<npts;iPt++)
161  for(int xyz=0;xyz<3;xyz++)
162  cp[xyz] += this->mesh->GetPoint(pts[iPt])[xyz]-bounds[xyz*2+0];
163  for(int xyz=0;xyz<3;xyz++)
164  cp[xyz] /= npts;
165  for(int iOverlay=0;iOverlay<(int)this->initial_pattern_generator.size();iOverlay++)
166  {
167  Overlay* overlay = this->initial_pattern_generator[iOverlay];
168 
169  int iC = overlay->GetTargetChemical();
170  if(iC<0 || iC>=this->GetNumberOfChemicals())
171  continue; // best for now to silently ignore this overlay, because the user has no way of editing the overlays (short of editing the file)
172  //throw runtime_error("Overlay: chemical out of range: "+GetChemicalName(iC));
173 
174  float val;
175  vector<float> vals(this->GetNumberOfChemicals());
176  for(int i=0;i<this->GetNumberOfChemicals();i++)
177  {
178  vtkFloatArray *scalars = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(i).c_str()) );
179  if(!scalars)
180  throw runtime_error("MeshRD::GenerateInitialPattern : failed to cast cell data to vtkFloatArray");
181  vals[i] = scalars->GetValue(iCell);
182  if(i==iC) val = vals[i];
183  }
184  vtkFloatArray *scalars = vtkFloatArray::SafeDownCast( this->mesh->GetCellData()->GetArray(GetChemicalName(iC).c_str()) );
185  if(!scalars)
186  throw runtime_error("MeshRD::GenerateInitialPattern : failed to cast cell data to vtkFloatArray");
187  scalars->SetValue(iCell,overlay->Apply(vals,this,cp[0],cp[1],cp[2]));
188  }
189  }
190  this->mesh->Modified();
191  this->timesteps_taken = 0;
192 }

705 void MeshRD::ComputeCellNeighbors(TNeighborhood neighborhood_type,int range,TWeight weight_type)
706 {
707  // TODO: for now we treat LAPLACIAN weights the same as EQUAL weights, not sure what to do with this on arbitrary meshes
708  if(range!=1)
709  throw runtime_error("MeshRD::ComputeCellNeighbors : unsupported range");
710  if(!this->mesh->IsHomogeneous())
711  throw runtime_error("MeshRD::ComputeCellNeighbors : mixed cell types not supported");
712 
713  vtkSmartPointer<vtkIdList> ptIds = vtkSmartPointer<vtkIdList>::New();
714  vtkSmartPointer<vtkIdList> cellIds = vtkSmartPointer<vtkIdList>::New();
715  TNeighbor nbor;
716 
717  vector<vector<TNeighbor> > cell_neighbors; // the connectivity between cells; for each cell, what cells are its neighbors?
718  this->max_neighbors = 0;
719  for(vtkIdType iCell=0;iCell<this->mesh->GetNumberOfCells();iCell++)
720  {
721  vector<TNeighbor> neighbors;
722  this->mesh->GetCellPoints(iCell,ptIds);
723  vtkIdType npts = ptIds->GetNumberOfIds();
724  switch(neighborhood_type)
725  {
726  case VERTEX_NEIGHBORS: // neighbors share a vertex
727  {
728  vtkSmartPointer<vtkIdList> vertIds = vtkSmartPointer<vtkIdList>::New();
729  vertIds->SetNumberOfIds(1);
730  // first try to add neighbors that are also edge-neighbors of the previously added cell
731  size_t n_previously;
732  do {
733  n_previously = neighbors.size();
734  for(vtkIdType iPt=0;iPt<npts;iPt++)
735  {
736  vertIds->SetId(0,ptIds->GetId(iPt));
737  this->mesh->GetCellNeighbors(iCell,vertIds,cellIds);
738  for(vtkIdType iNeighbor=0;iNeighbor<cellIds->GetNumberOfIds();iNeighbor++)
739  {
740  nbor.iNeighbor = cellIds->GetId(iNeighbor);
741  switch(weight_type)
742  {
743  case EQUAL: nbor.weight = 1.0f; break;
744  case LAPLACIAN: nbor.weight = 1.0f; break;
745  default: throw runtime_error("MeshRD::ComputeCellNeighbors : unsupported weight type");
746  }
747  if(neighbors.empty() || IsEdgeNeighbor(this->mesh,neighbors.back().iNeighbor,nbor.iNeighbor))
748  add_if_new(neighbors,nbor);
749  }
750  }
751  } while(neighbors.size() > n_previously);
752  // add any remaining neighbors (in case mesh is non-manifold)
753  for(vtkIdType iPt=0;iPt<npts;iPt++)
754  {
755  vertIds->SetId(0,ptIds->GetId(iPt));
756  this->mesh->GetCellNeighbors(iCell,vertIds,cellIds);
757  for(vtkIdType iNeighbor=0;iNeighbor<cellIds->GetNumberOfIds();iNeighbor++)
758  {
759  nbor.iNeighbor = cellIds->GetId(iNeighbor);
760  switch(weight_type)
761  {
762  case EQUAL: nbor.weight = 1.0f; break;
763  case LAPLACIAN: nbor.weight = 1.0f; break;
764  default: throw runtime_error("MeshRD::ComputeCellNeighbors : unsupported weight type");
765  }
766  add_if_new(neighbors,nbor);
767  }
768  }
769  }
770  break;
771  case EDGE_NEIGHBORS: // neighbors share an edge
772  {
773  vtkCell* pCell = this->mesh->GetCell(iCell);
774  for(int iEdge=0;iEdge<pCell->GetNumberOfEdges();iEdge++)
775  {
776  vtkIdList *vertIds = pCell->GetEdge(iEdge)->GetPointIds();
777  this->mesh->GetCellNeighbors(iCell,vertIds,cellIds);
778  for(vtkIdType iNeighbor=0;iNeighbor<cellIds->GetNumberOfIds();iNeighbor++)
779  {
780  nbor.iNeighbor = cellIds->GetId(iNeighbor);
781  switch(weight_type)
782  {
783  case EQUAL: nbor.weight = 1.0f; break;
784  case LAPLACIAN: nbor.weight = 1.0f; break;
785  default: throw runtime_error("MeshRD::ComputeCellNeighbors : unsupported weight type");
786  }
787  add_if_new(neighbors,nbor);
788  }
789  }
790  }
791  break;
792  case FACE_NEIGHBORS:
793  {
794  vtkCell* pCell = this->mesh->GetCell(iCell);
795  for(int iEdge=0;iEdge<pCell->GetNumberOfFaces();iEdge++)
796  {
797  vtkIdList *vertIds = pCell->GetFace(iEdge)->GetPointIds();
798  this->mesh->GetCellNeighbors(iCell,vertIds,cellIds);
799  for(vtkIdType iNeighbor=0;iNeighbor<cellIds->GetNumberOfIds();iNeighbor++)
800  {
801  nbor.iNeighbor = cellIds->GetId(iNeighbor);
802  switch(weight_type)
803  {
804  case EQUAL: nbor.weight = 1.0f; break;
805  case LAPLACIAN: nbor.weight = 1.0f; break;
806  default: throw runtime_error("MeshRD::ComputeCellNeighbors : unsupported weight type");
807  }
808  add_if_new(neighbors,nbor);
809  }
810  }
811  }
812  break;
813  default: throw runtime_error("MeshRD::ComputeCellNeighbors : unsupported neighborhood type");
814  }
815  // normalize the weights for this cell
816  float weight_sum=0.0f;
817  for(int iN=0;iN<(int)neighbors.size();iN++)
818  weight_sum += neighbors[iN].weight;
819  weight_sum = max(weight_sum,1e-5f); // avoid div0
820  for(int iN=0;iN<(int)neighbors.size();iN++)
821  neighbors[iN].weight /= weight_sum;
822  // store this list of neighbors
823  cell_neighbors.push_back(neighbors);
824  if((int)neighbors.size()>this->max_neighbors)
825  this->max_neighbors = (int)neighbors.size();
826  this->max_neighbors = max(1,this->max_neighbors); // avoid error in case of unconnected cells or single cell
827  }
828 
829  // copy data to plain arrays
830  if(this->cell_neighbor_indices) delete []this->cell_neighbor_indices;
831  if(this->cell_neighbor_weights) delete []this->cell_neighbor_weights;
832  this->cell_neighbor_indices = new int[this->mesh->GetNumberOfCells()*this->max_neighbors];
833  this->cell_neighbor_weights = new float[this->mesh->GetNumberOfCells()*this->max_neighbors];
834  for(int i=0;i<this->mesh->GetNumberOfCells();i++)
835  {
836  for(int j=0;j<(int)cell_neighbors[i].size();j++)
837  {
838  int k = i*this->max_neighbors + j;
839  this->cell_neighbor_indices[k] = cell_neighbors[i][j].iNeighbor;
840  this->cell_neighbor_weights[k] = cell_neighbors[i][j].weight;
841  }
842  // fill any remaining slots with iCell,0.0
843  for(int j=(int)cell_neighbors[i].size();j<max_neighbors;j++)
844  {
845  int k = i*this->max_neighbors + j;
846  this->cell_neighbor_indices[k] = i;
847  this->cell_neighbor_weights[k] = 0.0f;
848  }
849  }
850 }
