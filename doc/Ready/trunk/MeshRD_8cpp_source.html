00467     {

00469         this->mesh->GetCellPoints(iCell,ptIds);
00470         vtkIdType npts = ptIds->GetNumberOfIds();

00472         {
00473             case VTK_POLYGON: // 2D face, neighbors share an edge
00474                 {
00475                     for(vtkIdType iPt=0;iPt<npts;iPt++)
00476                     {
00477                         vtkSmartPointer<vtkIdList> edgeIds = vtkSmartPointer<vtkIdList>::New();
00478                         edgeIds->SetNumberOfIds(2);
00479                         edgeIds->SetId(0,ptIds->GetId(iPt));
00480                         edgeIds->SetId(1,ptIds->GetId((iPt+1)%npts));
00481                         this->mesh->GetCellNeighbors(iCell,edgeIds,cellIds);
00482                         int nNeighbors = cellIds->GetNumberOfIds();
00483                         for(vtkIdType iNeighbor=0;iNeighbor<nNeighbors;iNeighbor++)
00484                         {
00485                             vtkIdType id = cellIds->GetId(iNeighbor);
00486                             neighbors.push_back(id);
00487                         }
00488                     }
00489                     this->cell_neighbors.push_back(neighbors);
00490                 }
00491                 break;
00492             case VTK_TETRA: // 3D cell, neighbors share a triangular face
00493                 {
00494                     for(vtkIdType iPt=0;iPt<npts;iPt++)
00495                     {
00496                         vtkSmartPointer<vtkIdList> faceIds = vtkSmartPointer<vtkIdList>::New();
00497                         faceIds->SetNumberOfIds(3);

00501                         this->mesh->GetCellNeighbors(iCell,faceIds,cellIds);
