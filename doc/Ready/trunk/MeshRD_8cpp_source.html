00410             }
00411             for(vtkIdType iCell=0;iCell<(int)this->cell_neighbors.size();iCell++)
00412             {
00413                 float val = source->GetValue(iCell);
00414                 for(vtkIdType iNeighbor=0;iNeighbor<(int)this->cell_neighbors[iCell].size();iNeighbor++)
00415                     val += source->GetValue(this->cell_neighbors[iCell][iNeighbor]);
00416                 val /= this->cell_neighbors[iCell].size() + 1;
00417                 target->SetValue(iCell,val);
00418             }
00419         }
00420     }

00490                             vtkIdType id = cellIds->GetId(iNeighbor);
