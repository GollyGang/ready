00557     }
00558     // N.B. we don't intend to support cells of mixed dimensionality in one mesh
00559 
00560     // copy data to plain arrays
00561     this->cell_neighbor_indices = new int[this->mesh->GetNumberOfCells()*this->max_neighbors];
00562     this->cell_neighbor_weights = new float[this->mesh->GetNumberOfCells()*this->max_neighbors];
00563     for(int i=0;i<this->mesh->GetNumberOfCells();i++)
00564     {
00565         for(int j=0;j<(int)cell_neighbors[i].size();j++)
00566         {
00567             int k = i*this->max_neighbors + j;

00569             this->cell_neighbor_weights[k] = cell_neighbors[i][j].diffusion_coefficient;
00570         }
00571         // fill any remaining slots with iCell,0.0
00572         for(int j=(int)cell_neighbors[i].size();j<max_neighbors;j++)
00573         {
00574             int k = i*this->max_neighbors + j;
00575             this->cell_neighbor_indices[k] = i;
00576             this->cell_neighbor_weights[k] = 0.0f;
00577         }
00578     }
