168  enum TWeight { EQUAL, EUCLIDEAN_DISTANCE, BOUNDARY_SIZE, RANGE, VERTEX_NEIGHBORS_RANGE,
169  EDGE_NEIGHBORS_RANGE, FACE_NEIGHBORS_RANGE, LAPLACIAN };
170  // (not intending to support all of these options immediately, but here for thought)
171  // EQUAL: all weights are 1
172  // EUCLIDEAN_DISTANCE: weights are scaled by the distance to each cell centroid
173  // BOUNDARY_SIZE: weights are scaled by the length/area of the shared geometry between two neighbors
174  // RANGE: weight is minimum TNeighborhood steps to get from central cell
175  // VERTEX_NEIGHBORS_RANGE: weight is minimum vertex-neighbors steps to get from central cell
176  // EDGE_NEIGHBORS_RANGE: weight is minimum edge-neighbors steps to get from central cell
177  // FACE_NEIGHBORS_RANGE: weight is minimum face-neighbors steps to get from central cell
